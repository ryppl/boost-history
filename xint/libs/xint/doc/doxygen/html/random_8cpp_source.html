00115 bool generator_t::seed_secure() {
00116     const int cBitsRequested=256,
00117         cBytesRequested=cBitsRequested / std::numeric_limits<char>::digits;
00118     bool success=false;
00119 
00120     #ifdef _WIN32
00121         // This should work under WinXP, Vista, and Win7. No guarantees about
00122         // future compatibility, but I doubt that Microsoft will get rid of it
00123         // (it's too useful), and I doubt that they'll change it now that it's
00124         // well-known (it would break too many programs). We could also use the
00125         // rand_s function in more recent versions of Visual C++, but that
00126         // causes compatibility problems with older versions of Windows.
00127         typedef BOOLEAN (WINAPI *RtlGenRandomFn)(PVOID, ULONG);
00128         HMODULE dll=LoadLibrary(_T("Advapi32.dll"));
00129         if (dll != 0) {
00130             RtlGenRandomFn RtlGenRandom=RtlGenRandomFn(GetProcAddress(dll,
00131                 "SystemFunction036"));
00132             if (RtlGenRandom != 0) {
00133                 std::vector<char> buffer(cBytesRequested, '\0');
00134                 if (RtlGenRandom(&buffer[0], cBytesRequested)) {
00135                     char *c=&buffer[0];
00136                     seed_manual(std::string(c, c+cBytesRequested));
00137                     success=true;
00138                 }
00139             }
00140             FreeLibrary(dll);
00141         }
00142     #else
00143         // This should be supported under most non-Windows systems. Note that
00144         // we're using /dev/urandom, not /dev/random -- /dev/random is more
00145         // secure, but it can be VERY slow.
00146         std::ifstream rng("/dev/urandom");
00147         if (rng) {
00148             std::string rstr;
00149             for (int i=0; i < cBytesRequested; ++i)
00150                 rstr.push_back(rng.get());
00151             seed_manual(rstr);
00152             success=true;
00153         }
00154     #endif
00155 
00156     return success;
00157 }

00197 // (1<<bits)-1, inclusive
00198 integer random_by_size(size_t bits, bool highBitOn, bool lowBitOn, bool canBeNegative) {
00199     if (bits<=0) return integer::zero();
00200 
00201     generator_t randomGenerator;
00202 
00203     const size_t cBitsPerIteration=std::numeric_limits<generator_t::result_type>::digits;
00204 
00205     // Grab a set of random bits, of at least the specified size
00206     int iterations = (bits+cBitsPerIteration-1) / cBitsPerIteration;
00207     std::vector<generator_t::result_type> v;
00208     for (int i=0; i<iterations; ++i) v.push_back(randomGenerator());
00209 
00210     const char *vptr=(const char *)&v[0], *vptr_end=vptr+(v.size() * sizeof(generator_t::result_type));
00211     integer p=from_binary(std::string(vptr, vptr_end));
00212 
00213     // Trim it to the proper length
00214     int index=(bits/bits_per_digit);
00215     digit_t mask=(digit_t(1) << (bits % bits_per_digit))-1;
00216     if (mask==0) { mask=digit_mask; --index; }
00217     p._get_data()->digits[index] &= mask;
00218     for (digit_t *i=p._get_data()->digits+index+1,
00219         *ie=p._get_data()->digits+p._get_data()->mLength; i<ie; ++i) *i=0;
00220     p._get_data()->skipLeadingZeros();
00221 
00222     if (highBitOn) setbit(p, bits-1);
00223     if (lowBitOn) setbit(p, 0);
00224     if (canBeNegative) p._set_negative(randomGenerator() & 0x01);
00225 
00226     return p;
00227 }

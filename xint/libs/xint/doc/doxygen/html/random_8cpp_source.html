00179 // Returns a positive (unless told otherwise) integer between zero and
00180 // (1<<bits)-1, inclusive
00181 integer random_by_size(size_t bits, bool highBitOn, bool lowBitOn, bool
00182     canBeNegative)
00183 {
00184     if (bits<=0) return integer::zero();
00185 
00186     generator_t randomGenerator;
00187     const size_t cBitsPerIteration=std::numeric_limits<generator_t::result_type>::digits;
00188 
00189     // Grab a set of random bits, of at least the specified size
00190     size_t iterations = (bits+cBitsPerIteration-1) / cBitsPerIteration;
00191     std::vector<generator_t::result_type> v;
00192     for (size_t i=0; i<iterations; ++i) v.push_back(randomGenerator());
00193 
00194     const char *vptr=(const char *)&v[0], *vptr_end=vptr+(v.size() * sizeof(generator_t::result_type));
00195     integer p=from_binary(std::string(vptr, vptr_end));
00196 
00197     // Trim it to the proper length
00198     size_t index=(bits/bits_per_digit);
00199     digit_t mask=(digit_t(1) << (bits % bits_per_digit))-1;
00200     if (mask==0) { mask=digit_mask; --index; }
00201     p._get_data()->digits[index] &= mask;
00202     for (digit_t *i=p._get_data()->digits+index+1,
00203         *ie=p._get_data()->digits+p._get_data()->mLength; i<ie; ++i) *i=0;
00204     p._get_data()->skipLeadingZeros();
00205 
00206     if (highBitOn) setbit(p, bits-1);
00207     if (lowBitOn) setbit(p, 0);
00208     if (canBeNegative) p._set_negative(randomGenerator() & 0x01);
00209 
00210     return p;
00211 }

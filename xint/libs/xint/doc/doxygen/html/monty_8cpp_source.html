00024 digit_t inverse0(const integer& n) {
00025     // Using the Dussé and Kalisk simplification
00026     doubledigit_t x = 2, y = 1;
00027     digit_t n0 = n._get_digit(0);
00028     for (size_t i = 2; i <= bits_per_digit; ++i, x <<= 1)
00029         if (x < ((n0 * y) & ((x << 1) - 1)))
00030             y += x;
00031     return digit_t(x - y);
00032 }

00075 integer montgomeryMultiplyMod(const integer& a, const integer& b, const integer& n,
00076     digit_t nPrime0)
00077 {
00078     // Using the Dussé and Kalisk simplification
00079     // Unstated parameter B is a power of two representing the number of values
00080     //   that a single digit can hold, i.e. digit_overflowbit
00081     // Unstated parameter L is the number of digits in the modulus, i.e.
00082     //   n._get_length()
00083     // Unstated parameter r is B^L
00084     // nPrime0 is nPrime mod B, or digit zero of nPrime
00085 
00086     const integer B(digit_overflowbit);
00087     const size_t L(n._get_length()), L1(L-1);
00088 
00089     integer t=a*b;
00090     size_t i=0;
00091 
00092     do {
00093         digit_t mi=digit_t(doubledigit_t(t._get_digit(i))*nPrime0);
00094         t += (n * mi) << (bits_per_digit * i);
00095     } while (++i <= L1);
00096 
00097     t >>= (bits_per_digit * L); // Fast divide by r
00098     return (t >= n ? t - n : t);
00099 }

00155 int mostEfficientK(const integer& e) {
00156     doubledigit_t k=cMaxK, kTarget=log2(e)-1;
00157     while (k > 1 && ((k - 1) * (k << ((k - 1) << 1))
00158         / (ddPowerOfTwo(k) - k - 1)) >= kTarget)
00159             --k;
00160     return int(k);
00161 }

00182 vkbitdigit_t splitIntoKBitDigits(const integer& e, size_t k) {
00183     size_t eBits=log2(e), eDigits=(eBits+k-1)/k, i=0, bitsInHopper=0;
00184 
00185     vkbitdigit_t rval;
00186     rval.reserve(eDigits);
00187 
00188     doubledigit_t hopper=0, mask=(doubledigit_t(1)<<k)-1;
00189     while (eDigits-- > 0) {
00190         if (bitsInHopper < k && i < e._get_length()) {
00191             // Add more bits to the hopper
00192             hopper = hopper | (doubledigit_t(e._get_digit(i++)) << bitsInHopper);
00193             bitsInHopper += bits_per_digit;
00194         }
00195 
00196         // Grab k bits off the bottom
00197         if (bitsInHopper > 0) {
00198             rval.push_back(kbitdigit_t(hopper & mask));
00199             hopper >>= k;
00200             bitsInHopper-=k;
00201         } else {
00202             rval.push_back(0);
00203         }
00204     }

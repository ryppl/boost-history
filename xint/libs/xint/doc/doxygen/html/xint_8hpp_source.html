00345 template <typename T> integer::integer(const T& n,
00346     typename boost::enable_if<boost::is_integral<T> >::type*,
00347     typename boost::enable_if<boost::is_signed<T> >::type*)
00348 {
00349     if (n >= 0) {
00350         if (static_cast<T>(n & detail::digit_mask) == n)
00351             _init(detail::digit_t(n));
00352         else _init(boost::uintmax_t(n));
00353     } else if (n == (std::numeric_limits<T>::min)()) {
00354         // Have to treat the minimum number carefully, because -n is not
00355         // what you'd think it is.
00356         _init(boost::uintmax_t(-(n+1)));
00357         _set_negative(true);
00358         --(*this);
00359     } else {
00360         _init(boost::uintmax_t(-n));
00361         _set_negative(true);
00362     }

00374 template <typename T>
00375 T to(const integer& n) {
00376     n._throw_if_nan();
00377     if (n < (std::numeric_limits<T>::min)()
00378         || n > (std::numeric_limits<T>::max)())
00379             throw too_big("value out of range for requested conversion");
00380 
00381     T rval=0;
00382     size_t len=n._get_length();
00383     for (size_t x=0; x<len; ++x)
00384         rval=static_cast<T>((rval * detail::digit_overflowbit)
00385             + n._get_digit(len-x-1));
00386     if (n.sign() < 0) rval *= -1;
00387     return rval;
00388 }

00425     traits>& in, xint::integer& n)
00426 {

00428     int dec=(in.flags() & std::ios::dec) ? 1 : 0;
00429     int oct=(in.flags() & std::ios::oct) ? 1 : 0;
00430     int count=hex+dec+oct;
00431 
00432     size_t base=autobase;
00433     if (count==1) {
00434         if (hex) base=16;
00435         else if (oct) base=8;
00436         else base=10;
00437     } else if (count>1) base=10;
00438     // else auto-base
00439 
00440     std::string s;
00441     if (in.peek()=='+') {
00442         in.ignore();
00443     } else if (in.peek()=='-') {
00444         in.ignore();
00445         s.push_back('-');
00446     }
00447 
00448     if (base==autobase) {
00449         if (in.peek()=='0') {
00450             in.ignore();
00451             int c=in.peek();
00452             if (c=='x' || c=='X') {
00453                 in.ignore();
00454                 base=16;
00455             } else base=8;
00456         } else base=10;
00457     }
00458 
00459     if (in.peek()=='#') {
00460         // Must be either #NaN# or an error
00461         char buffer[6];
00462         std::streamsize size=in.readsome(buffer, 5);
00463         buffer[size]=0;
00464         std::string str(buffer);
00465 
00466         if (str==detail::nan_text) n=integer(not_a_number());
00467         else in.setstate(std::ios::failbit);
00468     } else {
00469         while (in) {
00470             int c=in.peek();

00472                 (base==10 && (c>='0' && c<='9')) ||

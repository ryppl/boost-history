00319 integer::integer(const T& n) {
00320     if (std::numeric_limits<T>::is_signed) {
00321         if (n >= 0) {
00322             if (static_cast<T>(n & detail::digit_mask) == n)

00342 template <typename T>
00343 T to(const integer& n) {
00344     n._throw_if_nan();
00345     if (n < (std::numeric_limits<T>::min)()
00346         || n > (std::numeric_limits<T>::max)())
00347             throw too_big("value out of range for requested conversion");
00348 
00349     T rval=0;
00350     int len=n._get_length();
00351     for (int x=0; x<len; ++x)
00352         rval=static_cast<T>((rval * detail::digit_overflowbit)
00353             + n._get_digit(len-x-1));
00354     if (n.sign() < 0) rval *= -1;
00355     return rval;

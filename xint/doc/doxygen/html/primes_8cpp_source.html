00048 // The Miller-Rabin Probabilistic Test of Primality
00049 int isProbablePrimeBaseB(const integer& n, const integer &b, callback_t
00050     callback)
00051 {
00052     const integer nMinus1(n - integer::one());
00053 
00054     // Find r and a satisfying: n-1=2^a * r, r odd
00055     integer r=nMinus1;
00056     unsigned long a=0;
00057     while (r.even()) { r >>= 1; ++a; }
00058 
00059     // We check the congruence class of b^((2^i)r) % n for each i
00060     // from 0 to a - 1. If any one is correct, then n passes.
00061     // Otherwise, n fails.
00062     integer test=powmod(b, r, n);
00063     if (test==integer::one() || test==nMinus1) return 1; // Better than 3/4 chance it's prime
00064 
00065     while (a-->0) {
00066         test=sqrmod(test, n);
00067         if (test==nMinus1) return 1;
00068     }
00069 
00070     if (callback && !callback()) return -1;
00071     return 0;
00072 }

00076 int is_prime(const integer& n, callback_t callback) {
00077     if (n < 2) {
00078         if (exceptions_allowed()) throw std::invalid_argument("xint::is_prime cannot test numbers below 2");
00079         else return -1;
00080     }
00081 
00082     // First we trial-divide it by the primes below 2000
00083     static const std::vector<int> cLowPrimes(sieveOfEratosthenes(2000));
00084     std::vector<int>::const_iterator i=cLowPrimes.begin(), ie=cLowPrimes.end();
00085     for (; i!=ie; ++i) if ((n % *i)==0) return (n==*i);
00086 
00087     // Run the number through the Miller-Rabin Probabilistic Test of Primality
00088     // a few times to see if it's actually (probably) prime.
00089     for (int count=0; count<5; ++count) {
00090         unsigned int k=detail::get_random();
00091         int isP=isProbablePrimeBaseB(n, abs(k), callback);
00092         if (isP <= 0) return isP;
00093     }
00094     return 1; // Appears to be prime!
00095 }

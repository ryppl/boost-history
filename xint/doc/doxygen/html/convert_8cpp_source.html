00042 std::string to_string(const integer& n, size_t base, bool upperCase) {
00043     if (n.nan()) return detail::nan_text;
00044     if (base<2 || base>36) {
00045         if (exceptions_allowed()) throw invalid_base();
00046         else return std::string();
00047     }
00048 
00049     if (n.sign()==0) return "0";
00050 
00051     std::ostringstream out;
00052     if (base==16) {
00053         // Special no-division version, primarily for debugging division
00054         const data_t *ndata=n._get_data();

00056             *lastDigit=firstDigit + ndata->mLength - 1;
00057 
00058         const int bitsPerDigit=4;
00059         const digit_t mask=(doubledigit_t(1) << bitsPerDigit)-1;
00060 
00061         // Suppress any leading zeros
00062         const digit_t *d=lastDigit;
00063         int digitShift=(bits_per_digit - bitsPerDigit);
00064         while (digitShift >= 0 && ((*d >> digitShift) & mask) == 0)
00065             digitShift -= bitsPerDigit;
00066 
00067         do {
00068             while (digitShift >= 0) {
00069                 out << nToChar((*d >> digitShift) & mask, upperCase);
00070                 digitShift -= bitsPerDigit;
00071             }
00072 
00073             digitShift=(bits_per_digit - bitsPerDigit);
00074         } while (d-- != firstDigit);
00075 
00076         std::string r(n.sign() < 0 ? std::string("-")+out.str() : out.str());
00077         return r;
00078     } else {
00079         const integer shift=base;
00080         std::pair<integer, integer> a=std::make_pair(n, integer::zero());
00081         a.first._set_negative(false);
00082 
00083         integer r;
00084         while (a.first.sign()!=0) {
00085             a=divide_r(a.first, shift);
00086             out << nToChar(a.second._get_digit(0), upperCase);
00087         }
00088 
00089         if (n.sign() < 0) out << '-';
00090 
00091         std::string rval=out.str();
00092         std::reverse(rval.begin(), rval.end());

00181         digit_t d(*p++);
00182         for (size_t i=0; i<bytesPerDigit; ++i) {
00183             temp.push_back((unsigned char)(d));

Describes an allocator that allocates portions of fixed size memory buffer (shared memory, mapped file...)

namespace boost {
  namespace shmem {
    template<typename SegmentManager> class allocator<void, SegmentManager>;

    template<typename T, typename SegmentManager> class allocator;

    template<typename T, typename SegmentManager> 
      struct allocator<T, SegmentManager >>;
    template<typename T, typename SegmentManager> 
      bool operator==(const allocator< T, SegmentManager > &, 
                      const allocator< T, SegmentManager > &);
    template<typename T, typename SegmentManager> 
      bool operator!=(const allocator< T, SegmentManager > &, 
                      const allocator< T, SegmentManager > &);
    template<typename T, typename A> 
      bool operator==(const cached_node_allocator< T, A > &, 
                      const cached_node_allocator< T, A > &);
    template<typename T, typename A> 
      bool operator!=(const cached_node_allocator< T, A > &, 
                      const cached_node_allocator< T, A > &);
    template<typename T, std::size_t N, typename A> 
      bool operator==(const node_allocator< T, N, A > &, 
                      const node_allocator< T, N, A > &);
    template<typename T, std::size_t N, typename A> 
      bool operator!=(const node_allocator< T, N, A > &, 
                      const node_allocator< T, N, A > &);
    template<typename T, std::size_t N, typename A> 
      bool operator==(const private_node_allocator< T, N, A > &, 
                      const private_node_allocator< T, N, A > &);
    template<typename T, std::size_t N, typename A> 
      bool operator!=(const private_node_allocator< T, N, A > &, 
                      const private_node_allocator< T, N, A > &);
  }
}

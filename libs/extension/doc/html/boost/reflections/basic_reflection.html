Class template basic_reflection

boost::reflections::basic_reflection —

Synopsis

template<typename Info = std::string, typename ParameterInfo = void, 
         typename TypeInfo = extensions::default_type_info> 
class basic_reflection {
public:
  // types
  typedef void ParameterInfo;

  // Internal class for reflecting constructors, functions etc.   template<typename T> 
  class reflector {
  public:
    // types
    typedef basic_function_info< Info, TypeInfo, ParameterInfo > function_info;   
    typedef basic_constructor_info< TypeInfo, ParameterInfo >    constructor_info;

    // construct/copy/destruct
    reflector(basic_reflection< Info, ParameterInfo, TypeInfo > *);

    // public member functions
    template<typename Params...> reflector & constructor() ;
    template<typename Data> reflector & data(Data T::*, Info) ;
    template<typename ReturnValue = void, typename Params...> 
      reflector & function(void(T::*)(), Info) ;
  };

  // public member functions
  template<typename Params...> 
    instance_constructor< Params...> get_constructor() const;
  template<typename Data> data< Data > get_data(Info) const;
  template<typename T> reflector< T > reflect() ;
};

Description

basic_reflection public member functions

  1. template<typename Params...> 
  instance_constructor< Params...> get_constructor() const;

    Attempt to retrieve any constructor whose Params match the list in Params... For example: 

     r.reflect<MyClass>()
  .constructor() ;
 instance_constructor<> ic = r.get_constructor();
 if (i.valid()) {
   instance i = ic();
 }

    Requires:

    None.

    Postconditions:

    None.

    Returns:

    A constructor reference (that must be checked for validity).

    		 
  2. template<typename Data> data< Data > get_data(Info info) const;

    Attempt to retrieve any data of type Data and the same Info. For example: 

     r.reflect<MyClass>()
  .constructor()
  .data(&MyClass:someInt, "My Int");
 instance_constructor<> ic = r.get_constructor();
 instance i = ic();
 data<int> d = r.get_data<int>("My Int");
 int& myIntRef = d(i);

    Requires:

    None.

    Postconditions:

    None.

    Returns:

    A data reference (that must be checked for validity).

    		 
  3. template<typename T> reflector< T > reflect() ;

    Requires:

    reflect() has not been called.

    Postconditions:

    None.

    Returns:

    A reflector instance.

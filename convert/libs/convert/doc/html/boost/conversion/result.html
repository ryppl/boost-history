Struct template result

boost::conversion::result — Encapsulation of the conversion result.

Synopsis

// In header: <boost/convert/api.hpp>

template<typename TypeOut> 
struct result {
  // types
  typedef boost::safebool< result >   safebool;         
  typedef TypeOut                     out_type;         
  typedef boost::optional< out_type > optional_out_type;

  enum status { good, bad, throw_bad };

  // construct/copy/destruct
  result(out_type const &, status);
  result(optional_out_type const &, status);

  // public member functions
  operator typename safebool::type() const;
  out_type const & value() const;
};

Description

result public construct/copy/destruct

  1. result(out_type const & v, status s);
  2. result(optional_out_type const & v, status s);

result public member functions

  1. operator typename safebool::type() const;

    Implicit conversion to safe-bool to allow checking the success of the conversion. For example, 

         convert<foo>::result result = convert<foo>::from(str, fallback_value);
     if ( result) conversion succeeded
     if (!result) conversion failed


  2. out_type const & value() const;

    Retrieve the actual result (value) of the conversion. For example, 

         convert<foo_type>::result result = convert<foo_type>::from(str, foo_value);
     if (result)
     {
         foo_type value = result.value();  // Safe to retrieve the result
     }

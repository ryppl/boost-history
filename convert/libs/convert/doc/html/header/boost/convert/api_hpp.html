Header <boost/convert/api.hpp>

Defines the Boost.Convert interface 

namespace boost {
  template<typename TypeOut, typename EnableOut> struct convert;
  namespace conversion {
    template<typename TypeOut> struct result;
    namespace parameter {

      // Specify throw-on-failure behavior explicitly. 
       BOOST_PARAMETER_KEYWORD(type, throw_);

      // Specify locale. 
       BOOST_PARAMETER_KEYWORD(type, locale_);

      // Provide fallback value. 
       BOOST_PARAMETER_KEYWORD(type, fallback_);

      // Not used. Work in progress. 
       BOOST_PARAMETER_KEYWORD(type, format_);
    }
  }
}

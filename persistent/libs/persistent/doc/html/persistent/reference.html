Header <boost/persistent/basic_loc.hpp>

namespace boost {
  namespace persistent {
    template<typename T> struct is_locator;

    template<typename T, typename TxMgr> class basic_loc;

    // Equivalent to r.cast_static<Y>(). 
    template<typename Y, typename T, typename TxMgr> 
      basic_loc< Y, TxMgr > 
      static_pointer_cast(basic_loc< T, TxMgr > const & r);

    // Equivalent to r.cast_const<Y>();. 
    template<typename Y, typename T, typename TxMgr> 
      basic_loc< Y, TxMgr > 
      const_pointer_cast(basic_loc< T, TxMgr > const & r);

    // Equivalent to r.cast_dynamic<Y>(). 
    template<typename Y, typename T, typename TxMgr> 
      basic_loc< Y, TxMgr > 
      dynamic_pointer_cast(basic_loc< T, TxMgr > const & r);

    // Equivalent to a.swap(b). 
    template<typename T, typename TxMgr> 
      void swap(basic_loc< T, TxMgr > & a, basic_loc< T, TxMgr > & b);
  }
}

Header <boost/persistent/interior_ptr.hpp>

namespace boost {
  namespace persistent {
    template<typename T> class interior_ptr;

    // Equivalent to a.swap(b). 
    template<typename T> 
      void swap(interior_ptr< T > & a, interior_ptr< T > & b);

    // Equivalent to p.get(). 
    template<typename T> T * get_pointer(interior_ptr< T > const & p);

    // Equivalent to p.cast_static<Y>(). 
    template<typename Y, typename T> 
      interior_ptr< Y > static_pointer_cast(interior_ptr< T > const & p);

    // Equivalent to p.cast_dynamic<Y>(). 
    template<typename Y, typename T> 
      interior_ptr< Y > dynamic_pointer_cast(interior_ptr< T > const & p);

    // Equivalent to p.cast_const<Y>(). 
    template<typename Y, typename T> 
      interior_ptr< Y > const_pointer_cast(interior_ptr< T > const & p);

    // Equivalent to os << p.get(). 
    template<typename E, typename T, typename Y> 
      std::basic_ostream< E, T > & 
      operator<<(std::basic_ostream< E, T > & os, interior_ptr< Y > const & p);
  }
}

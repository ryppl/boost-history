Class template endian_view

boost::integer::endian_view

Synopsis

// In header: <boost/integer/endian/endian_view.hpp>

template<typename Endian, typename T, 
         bool IsFundamental = is_fundamental<T>::value, 
         bool IsSeq = fusion::traits::is_sequence<T>::value> 
class endian_view {
public:
  // construct/copy/destruct
  endian_view(value_type &);
  endian_view& operator=(value_type);
  endian_view& operator=(endian_view const &);
  template<typename Endian2> 
    endian_view& operator=(endian_view< Endian2, T, true, false > const &);

  // private member functions
  operator value_type() const;
};

Description

endian_view public construct/copy/destruct

  1. endian_view(value_type & ref);

    Effects: Constructs an object of type endian_view<E, T, n_bits, A>. 

  2. endian_view& operator=(value_type val);

    value_type(this->ref_) == v.

    Returns: *this. 

  3. endian_view& operator=(endian_view const & rhs);

    this->ref_ == rhs.ref_.

    Returns: *this. 

  4. template<typename Endian2> 
  endian_view& operator=(endian_view< Endian2, T, true, false > const & rhs);

    value_type(this->ref_) == value_type(rhs.ref_).

    Returns: *this.

endian_view private member functions

  1. operator value_type() const;

    Returns: The converted value_type of the referenced type as it was seen as an endian aware type.

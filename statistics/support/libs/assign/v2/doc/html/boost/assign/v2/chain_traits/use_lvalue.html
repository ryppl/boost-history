template<typename U1, typename U2, typename Tag = use_default> 
struct use_lvalue {
  // types
  typedef boost::range_reference< U1 >::type                                                                 r1_;           
  typedef boost::range_reference< U2 >::type                                                                 r2_;           
  typedef boost::mpl::and_< boost::is_reference< r1_ >, boost::is_reference< r2_ > >                         are_refs_;     
  typedef boost::is_same< r1_, r2_ >                                                                         are_same_;     
  typedef boost::mpl::and_< are_refs_, are_same_ >                                                           are_same_refs_;
  typedef boost::is_const< typename boost::remove_reference< r1_ >::type >                                   is_const_;     
  typedef boost::mpl::eval_if< is_const_, boost::mpl::identity< boost::mpl::false_ >, are_same_refs_ >::type type;          

  // public member functions
   BOOST_STATIC_CONSTANT(bool, value = type::value) ;
};

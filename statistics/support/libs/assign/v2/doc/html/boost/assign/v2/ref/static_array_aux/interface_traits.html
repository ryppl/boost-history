// In header: <boost/assign/v2/ref/static_array/interface.hpp>

template<std::size_t N, typename Tag, typename T> 
struct interface_traits {
  // types
  typedef Tag                                                assign_tag_;    
  typedef ref::wrapper< assign_tag_, T >                     wrapper_;       
  typedef boost::array< wrapper_, N >                        impl_;          
  typedef ref::result_of::unwrap_range< impl_ >::type        range_;         
  typedef ref::result_of::unwrap_range< impl_ const  >::type range_c_;       
  typedef boost::unwrap_reference< T >::type                 value_type;     
  typedef boost::add_reference< value_type >::type           reference;      
  typedef boost::call_traits< reference >::const_reference   const_reference;
  typedef boost::range_iterator< range_ >::type              iterator;       
  typedef boost::range_iterator< range_c_ >::type            const_iterator; 
  typedef boost::range_size< impl_ >::type                   size_type;      
  typedef boost::range_difference< range_ >::type            difference_type;
};

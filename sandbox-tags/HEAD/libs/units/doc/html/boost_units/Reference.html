Header <boost/units/cmath.hpp>

Overloads of functions in <cmath> for quantities.

Header <boost/units/operators.hpp>

Compile time operators and typeof helper classes.

These operators declare the compile-time operators needed to support dimensional analysis algebra. Specializations must be defined for all desired operand types. Typeof helper classes define result type for heterogeneous operators on value types. These must be defined through specialization for powers and roots.

Header <boost/units/systems/physical_units.hpp>

Various definitions to simplify systems of physical units.

This file includes the seven SI-defined fundamental dimensions as well as composite dimensions for a number of commonly encountered physical unit types. Angles and solid angles are technically dimensionless. However, incorrect use of degrees and radians is a common source of errors, so we treat them on the same footing as the seven physical dimensions. 

namespace boost {
  namespace units {
    struct length_tag;
    struct mass_tag;
    struct time_tag;
    struct current_tag;
    struct temperature_tag;
    struct amount_tag;
    struct intensity_tag;
    struct angle_tag;
    struct solid_angle_tag;

    typedef fundamental_dimension< length_tag >::type length_type;  // fundamental dimension of length (L) 
    typedef fundamental_dimension< mass_tag >::type mass_type;  // fundamental dimension of mass (M) 
    typedef fundamental_dimension< time_tag >::type time_type;  // fundamental dimension of time (T) 
    typedef fundamental_dimension< current_tag >::type current_type;  // fundamental dimension of electric current (C) 
    typedef fundamental_dimension< temperature_tag >::type temperature_type;  // fundamental dimension of temperature (K) 
    typedef fundamental_dimension< amount_tag >::type amount_type;  // fundamental dimension of amount of substance (A) 
    typedef fundamental_dimension< intensity_tag >::type intensity_type;  // fundamental dimension of luminous intensity (I, deprecated by SI) 
    typedef fundamental_dimension< angle_tag >::type angle_type;  // fundamental dimension of planar angle (QP) 
    typedef fundamental_dimension< solid_angle_tag >::type solid_angle_type;  // fundamental dimension of solid angle (QS) 
    typedef composite_dimension< length_tag, 1, time_tag,-2 >::type acceleration_type;  // composite dimension for acceleration : L T^-2 
    typedef composite_dimension< length_tag, 2 >::type area_type;  // composite dimension for area : L^2 
    typedef composite_dimension< length_tag, 2, mass_tag, 1, time_tag,-2 >::type energy_type;  // composite dimension for energy : L^2 M T^-2 
    typedef composite_dimension< length_tag, 1, mass_tag, 1, time_tag,-2 >::type force_type;  // composite dimension for force : L M T^-2 
    typedef composite_dimension< time_tag,-1 >::type frequency_type;  // composite dimension for frequency : T^-1 
    typedef composite_dimension< length_tag,-3, mass_tag, 1 >::type mass_density_type;  // composite dimension for mass density : L^-3 M 
    typedef composite_dimension< length_tag, 1, mass_tag, 1, time_tag,-1 >::type momentum_type;  // composite dimension for linear momentum : L M T^-1 
    typedef composite_dimension< length_tag, 2, mass_tag, 1, time_tag,-3 >::type power_type;  // composite dimension for power : L^2 M T^-3 
    typedef composite_dimension< length_tag,-1, mass_tag, 1, time_tag,-2 >::type pressure_type;  // composite dimension for pressure : L^-1 M T^-2 
    typedef composite_dimension< length_tag,-1, mass_tag, 1, time_tag,-2 >::type stress_type;  // composite dimension for stress : L^-1 M T^-2 
    typedef composite_dimension< length_tag, 1, time_tag,-1 >::type velocity_type;  // composite dimension for velocity : L T^-1 
    typedef composite_dimension< length_tag, 3 >::type volume_type;  // composite dimension for volume : l^3 
    typedef composite_dimension< length_tag,-1 >::type wavenumber_type;  // composite dimension for wavenumber : L^-1 
  }

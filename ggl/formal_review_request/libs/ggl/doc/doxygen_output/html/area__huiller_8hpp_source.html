00085     static inline void apply(PointOfSegment const& p1,
00086                 PointOfSegment const& p2,
00087                 excess_sum& state)
00088     {
00089         if (! ggl::math::equals(get<0>(p1), get<0>(p2)))
00090         {
00091             // Distance p1 p2
00092             double a = state.distance_over_unit_sphere(p1, p2);
00093 
00094             // Sides on unit sphere to south pole
00095             double b = 0.5 * math::pi - ggl::get_as_radian<1>(p2);
00096             double c = 0.5 * math::pi - ggl::get_as_radian<1>(p1);
00097 
00098             // Semi parameter
00099             double s = 0.5 * (a + b + c);
00100 
00101             // E: spherical excess, using l'Huiller's formula
00102             // [tg(e / 4)]2   =   tg[s / 2]  tg[(s-a) / 2]  tg[(s-b) / 2]  tg[(s-c) / 2]
00103             double E = 4.0 * atan(sqrt(std::abs(tan(s / 2)
00104                     * tan((s - a) / 2)
00105                     * tan((s - b) / 2)
00106                     * tan((s - c) / 2))));
00107 
00108             E = std::abs(E);
00109 
00110             // In right direction: positive, add area. In left direction: negative, subtract area.
00111             // Longitude comparisons are not so obvious. If one is negative, other is positive,
00112             // we have to take the dateline into account.
00113             // TODO: check this / enhance this, should be more robust. See also the "grow" for ll
00114             // TODO: use minmax or "smaller"/"compare" strategy for this
00115             double lon1 = ggl::get_as_radian<0>(p1) < 0
00116                 ? ggl::get_as_radian<0>(p1) + math::two_pi
00117                 : ggl::get_as_radian<0>(p1);
00118 
00119             double lon2 = ggl::get_as_radian<0>(p2) < 0
00120                 ? ggl::get_as_radian<0>(p2) + math::two_pi
00121                 : ggl::get_as_radian<0>(p2);
00122 
00123             if (lon2 < lon1)
00124             {
00125                 E = -E;
00126             }
00127 
00128             state.sum += E;
00129         }
00130     }
00131

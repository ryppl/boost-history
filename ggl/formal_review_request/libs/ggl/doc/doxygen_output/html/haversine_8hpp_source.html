00069                 inline return_type calc(const T1& lon1, const T1& lat1, const T2& lon2, const T2& lat2) const
00070                 {
00071                     double a = math::hav(lat2 - lat1) + cos(lat1) * cos(lat2) * math::hav(lon2 - lon1);
00072                     double c = 2.0 * asin(sqrt(a));
00073                     return return_type(m_radius * c);

00132                 inline double course(PR const& p1, PR const& p2) const
00133                 {
00134                     /***
00135                         Course between points
00136 
00137                         We obtain the initial course, tc1, (at point 1) from point 1 to point 2 by the following. The formula fails if the initial point is a pole. We can special case this with:
00138 
00139                         IF (cos(lat1) < EPS)   // EPS a small number ~ machine precision
00140                           IF (lat1 > 0): tc1= pi        //  starting from N pole
00141                           ELSE: tc1= 2*pi         //  starting from S pole
00142                           ENDIF
00143                         ENDIF
00144 
00145                         For starting points other than the poles:
00146                         IF sin(lon2-lon1)<0: tc1=acos((sin(lat2)-sin(lat1)*cos(d))/(sin(d)*cos(lat1)))
00147                         ELSE: tc1=2*pi-acos((sin(lat2)-sin(lat1)*cos(d))/(sin(d)*cos(lat1)))
00148                         ENDIF
00149 
00150                         An alternative formula, not requiring the pre-computation of d, the distance between the points, is:
00151                            tc1=mod(atan2(sin(lon1-lon2)*cos(lat2),
00152                                    cos(lat1)*sin(lat2)-sin(lat1)*cos(lat2)*cos(lon1-lon2))
00153                                    , 2*pi)
00154                      ***/
00155                     double dlon = get<0>(p2) - get<0>(p1);
00156                     double cos_p2lat = cos(get<1>(p2));
00157                     return atan2(sin(dlon) * cos_p2lat,
00158                         cos(get<1>(p1)) * sin(get<1>(p2))
00159                         - sin(get<1>(p1)) * cos_p2lat * cos(dlon));
00160                 }

00162                 inline return_type calc(PR const& p, PR const& sp1, PR const& sp2) const
00163                 {
00164                     /***
00165                     Cross track error:
00166                     Suppose you are proceeding on a great circle route from A to B (course =crs_AB) and end up at D, perhaps off course.
00167                     (We presume that A is ot a pole!) You can calculate the course from A to D (crs_AD) and the distance from A to D (dist_AD)
00168                     using the formulae above. In shifteds of these the cross track error, XTD, (distance off course) is given by
00169 
00170                                XTD =asin(sin(dist_AD)*sin(crs_AD-crs_AB))
00171 
00172                     (positive XTD means right of course, negative means left)
00173                     (If the point A is the N. or S. Pole replace crs_AD-crs_AB with
00174                     lon_D-lon_B or lon_B-lon_D, respectively.)
00175                      ***/
00176 
00177                     // Calculate distances, in radians, on the unit sphere
00178                     // It seems not useful to let this strategy be templatized, it should be in radians and on the unit sphere
00179                     strategy::distance::haversine<PR, PR> strategy(1.0);
00180                     double d1 = strategy(sp1, p);
00181 
00182                     // Actually, calculation of d2 not necessary if we know that the projected point is on the great circle...
00183                     double d2 = strategy(sp2, p);
00184 
00185                     // Source: http://williams.best.vwh.net/avform.htm
00186 
00187                     double crs_AD = course(sp1, p);
00188                     double crs_AB = course(sp1, sp2);
00189                     double XTD = fabs(asin(sin(d1) * sin(crs_AD - crs_AB)));
00190 
00191                     // Return shortest distance, either to projected point on segment sp1-sp2, or to sp1, or to sp2
00192                     return return_type(m_radius * (std::min)((std::min)(d1, d2), XTD));
00193                 }

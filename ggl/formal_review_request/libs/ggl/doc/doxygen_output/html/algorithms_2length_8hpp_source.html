00062 template<typename Range, typename Strategy>
00063 struct range_length
00064 {
00065     static inline double apply(Range const& range, Strategy const& strategy)
00066     {
00067         typedef typename ggl::coordinate_type<Range>::type coordinate_type;
00068 
00069         // Because result is square-rooted, for integer, the cast should
00070         // go to double and NOT to T
00071         typedef typename
00072             boost::mpl::if_c
00073             <
00074                 boost::is_integral<coordinate_type>::type::value,
00075                 double,
00076                 coordinate_type
00077             >::type calculation_type;
00078 
00079         calculation_type sum = 0.0;
00080 
00081         typedef typename boost::range_const_iterator<Range>::type iterator_type;
00082         iterator_type it = boost::begin(range);
00083         if (it != boost::end(range))
00084         {
00085             iterator_type previous = it++;
00086             while(it != boost::end(range))
00087             {
00088                 // Add point-point distance using the return type belonging to strategy
00089                 sum += strategy(*previous, *it);
00090                 previous = it++;
00091             }
00092         }
00093 
00094         return sum;
00095     }

00082                 inline double course(P const& p1, P const& p2) const
00083                 {
00084                     // http://williams.best.vwh.net/avform.htm#Crs
00085                     double dlon = get_as_radian<0>(p2) - get_as_radian<0>(p1);
00086                     double cos_p2lat = cos(get_as_radian<1>(p2));
00087 
00088                     // "An alternative formula, not requiring the pre-computation of d"
00089                     return atan2(sin(dlon) * cos_p2lat,
00090                         cos(get_as_radian<1>(p1)) * sin(get_as_radian<1>(p2))
00091                         - sin(get_as_radian<1>(p1)) * cos_p2lat * cos(dlon));
00092                 }

00094                 inline return_type calc(P const& p, P const& sp1, P const& sp2) const
00095                 {
00096                     // http://williams.best.vwh.net/avform.htm#XTE
00097 
00098                     double d1 = m_strategy(sp1, p);
00099 
00100                     // Actually, calculation of d2 not necessary if we know that the projected point is on the great circle...
00101                     double d2 = m_strategy(sp2, p);
00102 
00103                     double crs_AD = course(sp1, p);
00104                     double crs_AB = course(sp1, sp2);
00105                     double XTD = std::abs(asin(sin(d1) * sin(crs_AD - crs_AB)));
00106 
00107 //std::cout << "Course " << dsv(sp1) << " to " << dsv(p) << " " << crs_AD * ggl::math::r2d << std::endl;

00110 
00111 
00112                     // Return shortest distance, either to projected point on segment sp1-sp2, or to sp1, or to sp2
00113                     return return_type(m_radius * (std::min)((std::min)(d1, d2), XTD));

00181 
00182         // Clip the segment. Five situations:
00183         // 1. Segment is invisible, finish line if any (shouldn't occur)
00184         // 2. Segment is completely visible. Add (p1)-p2 to line
00185         // 3. Point 1 is invisible (clipped), point 2 is visible. Start new line from p1-p2...
00186         // 4. Point 1 is visible, point 2 is invisible (clipped). End the line with ...p2
00187         // 5. Point 1 and point 2 are both invisible (clipped). Start/finish an independant line p1-p2
00188         //
00189         // This results in:
00190         // a. if p1 is clipped, start new line
00191         // b. if segment is partly or completely visible, add the segment
00192         // c. if p2 is clipped, end the line
00193 
00194         bool c1 = false;
00195         bool c2 = false;
00196         segment<point_type> s(p1, p2);
00197 
00198         if (!strategy.clip_segment(b, s, c1, c2))
00199         {
00200             strategy.add(line_out, out);
00201         }
00202         else
00203         {
00204             // a. If necessary, finish the line and add a start a new one
00205             if (c1)
00206             {
00207                 strategy.add(line_out, out);
00208             }
00209 
00210             // b. Add p1 only if it is the first point, then add p2
00211             if (boost::empty(line_out))

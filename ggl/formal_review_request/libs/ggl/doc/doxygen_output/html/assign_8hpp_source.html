00211     typedef typename coordinate_type<B>::type coordinate_type;

00213     // Here we assign 4 coordinates to a box.
00214     // -> Most logical is: x1,y1,x2,y2
00215     // In case the user reverses x1/x2 or y1/y2, we could reverse them (THAT IS NOT IMPLEMENTED)

00217     // Note also comment in util/assign_box_corner ->
00218     //   ("Most logical is LOWER, UPPER and sub-order LEFT, RIGHT")

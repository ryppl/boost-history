00269             // Line with one point ==> no sections

00284             previous = it++, i++)

00286             segment_type s(*previous, *it);
00287 
00288             int direction_classes[DimensionCount] = {0};
00289             get_direction_loop
00290                 <
00291                     segment_type, 0, DimensionCount
00292                 >::apply(s, direction_classes);
00293 
00294             // if "dir" == 0 for all point-dimensions, it is duplicate.
00295             // Those sections might be omitted, if wished, lateron
00296             bool check_duplicate = true; //?
00297             bool duplicate = false;
00298 
00299             if (check_duplicate && direction_classes[0] == 0)
00300             {
00301                 // Recheck because all dimensions should be checked,
00302                 // not only first one,
00303                 // Note that DimensionCount might be < dimension<P>::value
00304                 if (check_duplicate_loop
00305                     <
00306                         segment_type, 0, ggl::dimension<Point>::type::value
00307                     >::apply(s)
00308                     )
00309                 {
00310                     duplicate = true;

00312                     // Change direction-info to force new section
00313                     // Note that wo consecutive duplicate segments will generate
00314                     // only one duplicate-section.
00315                     // Actual value is not important as long as it is not -1,0,1
00316                     assign_loop
00317                     <
00318                         int, 0, DimensionCount
00319                     >::apply(direction_classes, -99);
00320                 }
00321             }

00324                 && (!compare_loop
00325                         <
00326                             int, 0, DimensionCount
00327                         >::apply(direction_classes, section.directions)
00328                     || section.count > MaxCount
00329                     )
00330                 )

00332                 sections.push_back(section);
00333                 section = sections_range_type();
00334             }
00335 
00336             if (section.count == 0)
00337             {
00338                 section.begin_index = i;
00339                 section.ring_index = ring_index;
00340                 section.multi_index = multi_index;
00341                 section.duplicate = duplicate;
00342                 section.non_duplicate_index = ndi;
00343                 section.range_count = boost::size(range);

00346                     <
00347                         int, 0, DimensionCount

00357                 ndi++;
00358             }
00359         }
00360

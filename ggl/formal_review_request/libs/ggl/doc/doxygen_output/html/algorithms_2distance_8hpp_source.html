00096     static inline return_type apply(P const& point, L const& linestring,

00098     {
00099         typedef segment<const typename point_type<L>::type> segment_type;
00100 
00101         if (boost::begin(linestring) == boost::end(linestring))
00102         {
00103             return return_type(0);
00104         }
00105 
00106         // line of one point: return point square_distance
00107         typedef typename boost::range_const_iterator<L>::type iterator_type;
00108         iterator_type it = boost::begin(linestring);
00109         iterator_type prev = it++;
00110         if (it == boost::end(linestring))
00111         {
00112             return pp_strategy(point, *boost::begin(linestring));
00113         }
00114 
00115 
00116         // start with first segment distance
00117         return_type d = ps_strategy(point, segment_type(*prev, *it));
00118 
00119         // check if other segments are closer
00120         prev = it++;
00121         while(it != boost::end(linestring))
00122         {
00123             return_type ds = ps_strategy(point, segment_type(*prev, *it));
00124             if (ggl::close_to_zero(ds))
00125             {
00126                 return return_type(0);
00127             }
00128             else if (ds < d)
00129             {
00130                 d = ds;
00131             }
00132             prev = it++;
00133         }
00134 
00135         return d;
00136     }

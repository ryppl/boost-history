00031 template <typename PointType>

00037     {
00038         coordinate_type const& left0 = ggl::get<0>(lhs);

00041         return math::equals(left0, right0)
00042             ? ggl::get<1>(lhs) < ggl::get<1>(rhs)

00063 {
00064     typedef typename boost::range_iterator<V>::type iterator_type;
00065     typedef typename boost::range_value<V>::type ip_type;

00070          it != boost::end(intersection_points);

00101                                 || tvit->other_id == tvit2->seg_id

00103                                 )
00104                             {

00169 template <typename IntersectionPoints>
00170 inline void merge_intersection_points(IntersectionPoints& intersection_points)
00171 {
00172     typedef typename boost::range_value<IntersectionPoints>::type trav_type;
00173 
00174     if (boost::size(intersection_points) <= 1)
00175     {
00176         return;
00177     }

00179 
00180     // Sort all IP's from left->right, ymin->ymax such that
00181     // all same IP's are consecutive

00184     // (otherwise indexes are wrong)
00185     std::sort(boost::begin(intersection_points),
00186         boost::end(intersection_points),
00187         detail::intersection::on_increasing_dimension<trav_type>());

00191 #ifdef GGL_DEBUG_INTERSECTION
00192     std::cout << "Sorted (x then y): " << std::endl;

00194         it != boost::end(intersection_points); ++it)
00195     {
00196         std::cout << *it;
00197     }
00198 #endif
00199     bool has_merge = false;
00200 
00201     // Merge all same IP's, combining there IP/segment-info entries
00202     iterator it = boost::begin(intersection_points);
00203     for (iterator prev = it++; it != boost::end(intersection_points); ++it)
00204     {
00205         // IP can be merged if the point is equal
00206         if (ggl::equals(prev->point, it->point))
00207         {
00208             has_merge = true;
00209             prev->shared = true;
00210             prev->trivial = false;
00211             it->flagged = true;
00212             std::copy(it->info.begin(), it->info.end(),
00213                         std::back_inserter(prev->info));
00214         }
00215         else
00216         {
00217             prev = it;
00218         }
00219     }
00220 
00221 
00222     if (has_merge)
00223     {
00224 #ifdef GGL_DEBUG_INTERSECTION
00225         std::cout << "Merged (1): " << std::endl;

00227 #endif
00228 
00229         // If there merges, there might be  collinearities
00230         detail::intersection::remove_collinearities(intersection_points);

00033             // left of segment (> 0), right of segment (< 0), on segment (0)
00034             // In fact this is twice the area of a triangle
00035             static inline typename select_coordinate_type<P, PS>::type
00036                 side(const segment<const PS>& s, const P& p)
00037             {
00038                 typedef typename select_coordinate_type<P, PS>::type T;
00039 
00040                 // Todo: might be changed to subtract_point
00041                 T dx = get<1, 0>(s) - get<0, 0>(s);
00042                 T dy = get<1, 1>(s) - get<0, 1>(s);
00043                 T dpx = get<0>(p) - get<0, 0>(s);
00044                 T dpy = get<1>(p) - get<0, 1>(s);
00045                 return dx * dpy - dy * dpx;
00046             }
00047

00050             {
00051                 typename coordinate_type<P>::type s = side(segment<const P>(p0, p1), p2);
00052                 return s > 0 ? 1 : s < 0 ? -1 : 0;
00053             }

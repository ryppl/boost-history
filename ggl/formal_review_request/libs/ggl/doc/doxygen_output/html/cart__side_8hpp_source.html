00028         struct xy_side

00030             typedef typename select_coordinate_type<P, PS>::type coordinate_type;
00031 
00032             // Check at which side of a segment a point lies:
00033             // left of segment (> 0), right of segment (< 0), on segment (0)
00034             // In fact this is twice the area of a triangle
00035             static inline coordinate_type
00036                 side(segment<const PS> const& s, P const& p)
00037             {
00038                 coordinate_type const x = get<0>(p);
00039                 coordinate_type const y = get<1>(p);
00040 
00041                 coordinate_type const sx1 = get<0, 0>(s);
00042                 coordinate_type const sy1 = get<0, 1>(s);
00043                 coordinate_type const sx2 = get<1, 0>(s);
00044                 coordinate_type const sy2 = get<1, 1>(s);
00045 
00046                 // Todo: might be changed to subtract_point
00047                 coordinate_type dx = sx2 - sx1;
00048                 coordinate_type dy = sy2 - sy1;
00049                 coordinate_type dpx = x - sx1;
00050                 coordinate_type dpy = y - sy1;
00051                 return dx * dpy - dy * dpx;
00052             }
00053 
00054 
00055             static inline int side(PS const& p0, PS const& p1, P const& p2)
00056             {
00057                 coordinate_type s = side(segment<const PS>(p0, p1), p2);
00058                 return s > 0 ? 1 : s < 0 ? -1 : 0;
00059             }
00060         };

00141 template<typename Ring, typename Point, typename Strategy>
00142 struct centroid_ring
00143 {
00144     static inline bool apply(Ring const& ring, Point& centroid,
00145             Strategy const& strategy, typename Strategy::state_type& state)
00146     {
00147         typedef typename boost::range_const_iterator<Ring>::type iterator_type;
00148 
00149         // There is one thing where centroid is different from e.g. within.
00150         // If the ring has only one point, it might make sense that
00151         // that point is the centroid. So then, pass it twice, as a trick
00152         // to avoid adding another method.
00153 
00154         switch(boost::size(ring))
00155         {
00156             case 0 : // will cause exception in the end
00157                 return true; 
00158             case 1 :
00159                 Strategy::apply(ring.front(), ring.front(), state);
00160                 return false;
00161         }
00162 
00163         // Normal behaviour:
00164 
00165         iterator_type it = boost::begin(ring);
00166         for (iterator_type previous = it++;
00167             it != boost::end(ring);
00168             previous = it++)
00169         {
00170             Strategy::apply(*previous, *it, state);
00171         }
00172         return true;

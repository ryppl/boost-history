00068 template<typename P, typename R>
00069 inline bool ring_ok(R const& ring, P& c)
00070 {
00071     std::size_t const n = boost::size(ring);
00072     if (n > 1)
00073     {
00074         return true;
00075     }
00076     else if (n <= 0)
00077     {
00078         throw centroid_exception();
00079     }
00080     else
00081     {
00082         // n == 1: Take over the first point in a "coordinate neutral way"
00083         copy_coordinates(ring.front(), c);
00084         return false;
00085     }
00086     return true;

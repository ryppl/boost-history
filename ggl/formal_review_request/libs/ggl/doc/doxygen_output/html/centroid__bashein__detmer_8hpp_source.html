00149     static inline void apply(PointOfSegment const& p1,
00150             PointOfSegment const& p2, sums& state)
00151     {
00152         /* Algorithm:
00153         For each segment:
00154         begin
00155             ai = x1 * y2 - x2 * y1;
00156             sum_a2 += ai;
00157             sum_x += ai * (x1 + x2);
00158             sum_y += ai * (y1 + y2);
00159         end
00160         return POINT(sum_x / (3 * sum_a2), sum_y / (3 * sum_a2) )
00161         */
00162 
00163         // Get coordinates and promote them to calculation_type
00164         calculation_type const x1 = boost::numeric_cast<calculation_type>(get<0>(p1));
00165         calculation_type const y1 = boost::numeric_cast<calculation_type>(get<1>(p1));
00166         calculation_type const x2 = boost::numeric_cast<calculation_type>(get<0>(p2));
00167         calculation_type const y2 = boost::numeric_cast<calculation_type>(get<1>(p2));
00168         calculation_type const ai = x1 * y2 - x2 * y1;
00169         state.count++;
00170         state.sum_a2 += ai;
00171         state.sum_x += ai * (x1 + x2);
00172         state.sum_y += ai * (y1 + y2);
00173     }

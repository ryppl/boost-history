00113 /* ======= Crossings Multiply algorithm =================================== */
00114 
00115 /*
00116  * This version is usually somewhat faster than the original published in
00117  * Graphics Gems IV; by turning the division for testing the X axis crossing
00118  * into a tricky multiplication test this part of the test became faster,
00119  * which had the additional effect of making the test for "both to left or
00120  * both to right" a bit slower for triangles than simply computing the
00121  * intersection each time.  The main increase is in triangle testing speed,
00122  * which was about 15% faster; all other polygon complexities were pretty much
00123  * the same as before.  On machines where division is very expensive (not the
00124  * case on the HP 9000 series on which I tested) this test should be much
00125  * faster overall than the old code.  Your mileage may (in fact, will) vary,
00126  * depending on the machine and the test data, but in general I believe this
00127  * code is both shorter and faster.  This test was inspired by unpublished
00128  * Graphics Gems submitted by Joseph Samosky and Mark Haigh-Hutchinson.
00129  * Related work by Samosky is in:
00130  *
00131  * Samosky, Joseph, "SectionView: A system for interactively specifying and
00132  * visualizing sections through three-dimensional medical image data",
00133  * M.S. Thesis, Department of Electrical Engineering and Computer Science,
00134  * Massachusetts Institute of Technology, 1993.
00135  *
00136  */
00137 
00138 /* Shoot a test ray along +X axis.  The strategy is to compare vertex Y values
00139  * to the testing point's Y and quickly discard edges which are entirely to one
00140  * side of the test ray.  Note that CONVEX and WINDING code can be added as
00141  * for the CrossingsTest() code; it is left out here for clarity.
00142  *
00143  * Input 2D polygon _pgon_ with _numverts_ number of vertices and test point
00144  * _point_, returns 1 if inside, 0 if outside.
00145  */
00146 int CrossingsMultiplyTest( pgon, numverts, point )
00147 double  pgon[][2] ;
00148 int numverts ;
00149 double  point[2] ;
00150 {
00151 register int    j, yflag0, yflag1, inside_flag ;
00152 register double ty, tx, *vtx0, *vtx1 ;
00153 
00154     tx = point[X] ;
00155     ty = point[Y] ;
00156 
00157     vtx0 = pgon[numverts-1] ;
00158     /* get test bit for above/below X axis */
00159     yflag0 = ( vtx0[Y] >= ty ) ;
00160     vtx1 = pgon[0] ;
00161 
00162     inside_flag = 0 ;
00163     for ( j = numverts+1 ; --j ; ) {
00164 
00165     yflag1 = ( vtx1[Y] >= ty ) ;
00166     /* Check if endpoints straddle (are on opposite sides) of X axis
00167      * (i.e. the Y's differ); if so, +X ray could intersect this edge.
00168      * The old test also checked whether the endpoints are both to the
00169      * right or to the left of the test point.  However, given the faster
00170      * intersection point computation used below, this test was found to
00171      * be a break-even proposition for most polygons and a loser for
00172      * triangles (where 50% or more of the edges which survive this test
00173      * will cross quadrants and so have to have the X intersection computed
00174      * anyway).  I credit Joseph Samosky with inspiring me to try dropping
00175      * the "both left or both right" part of my code.
00176      */
00177     if ( yflag0 != yflag1 ) {
00178         /* Check intersection of pgon segment with +X ray.
00179          * Note if >= point's X; if so, the ray hits it.
00180          * The division operation is avoided for the ">=" test by checking
00181          * the sign of the first vertex wrto the test point; idea inspired
00182          * by Joseph Samosky's and Mark Haigh-Hutchinson's different
00183          * polygon inclusion tests.
00184          */
00185         if ( ((vtx1[Y]-ty) * (vtx0[X]-vtx1[X]) >=
00186             (vtx1[X]-tx) * (vtx0[Y]-vtx1[Y])) == yflag1 ) {
00187         inside_flag = !inside_flag ;
00188         }
00189     }
00190 
00191     /* Move to the next pair of vertices, retaining info as possible. */
00192     yflag0 = yflag1 ;
00193     vtx0 = vtx1 ;
00194     vtx1 += 2 ;
00195     }
00196 
00197     return( inside_flag ) ;
00198 }

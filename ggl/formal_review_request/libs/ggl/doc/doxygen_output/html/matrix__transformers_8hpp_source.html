00053 template <typename P1, typename P2>
00054 class ublas_transformer<P1, P2, 2, 2>
00055 {
00056 protected :
00057     typedef typename select_coordinate_type<P1, P2>::type coordinate_type;
00058     typedef coordinate_type ct; // Abbreviation
00059     typedef boost::numeric::ublas::matrix<coordinate_type> matrix_type;
00060     matrix_type m_matrix;
00061 
00062 public :
00063     inline ublas_transformer(
00064                 ct const& m_0_0, ct const& m_0_1, ct const& m_0_2,
00065                 ct const& m_1_0, ct const& m_1_1, ct const& m_1_2,
00066                 ct const& m_2_0, ct const& m_2_1, ct const& m_2_2)
00067         : m_matrix(3, 3)
00068     {
00069         m_matrix(0,0) = m_0_0;   m_matrix(0,1) = m_0_1;   m_matrix(0,2) = m_0_2;
00070         m_matrix(1,0) = m_1_0;   m_matrix(1,1) = m_1_1;   m_matrix(1,2) = m_1_2;
00071         m_matrix(2,0) = m_2_0;   m_matrix(2,1) = m_2_1;   m_matrix(2,2) = m_2_2;
00072     }
00073 
00074     inline ublas_transformer()
00075         : m_matrix(3, 3)
00076     {
00077     }
00078 
00079 
00080     inline bool operator()(const P1& p1, P2& p2) const
00081     {
00082         assert_dimension_greater_equal<P1, 2>();
00083         assert_dimension_greater_equal<P2, 2>();
00084 
00085         const coordinate_type& c1 = get<0>(p1);
00086         const coordinate_type& c2 = get<1>(p1);
00087 
00088         typedef typename ggl::coordinate_type<P2>::type ct2;
00089 
00090         set<0>(p2, boost::numeric_cast<ct2>(c1 * m_matrix(0,0)
00091                 + c2 * m_matrix(0,1) + m_matrix(0,2)));
00092         set<1>(p2, boost::numeric_cast<ct2>(c1 * m_matrix(1,0)
00093                 + c2 * m_matrix(1,1) + m_matrix(1,2)));
00094 
00095         return true;
00096     }
00097 
00098     const matrix_type& matrix() const { return m_matrix; }
00099 };

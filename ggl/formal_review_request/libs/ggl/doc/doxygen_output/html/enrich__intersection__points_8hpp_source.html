00128 // Sorts on segment + ring_index + multi_index
00129 template <typename Indexed>
00130 struct sort_on_segment_identifier
00131 {
00132     inline bool operator()(Indexed const& left, Indexed const& right) const
00133     {
00134         // Sanity check
00135         BOOST_ASSERT (left.subject.info.size() > 0 && right.subject.info.size() > 0);
00136 
00137         segment_identifier const& sl = left.subject.info.front().seg_id;
00138         segment_identifier const& sr = right.subject.info.front().seg_id;
00139 
00140         return sl == sr
00141             ? left.subject.info.front().distance < right.subject.info.front().distance
00142             : sl < sr;

00221 // Creates selection of IP-s of only this unique source/segment,
00222 // then sorts on distance,
00223 // then assigns for each IP which is the next IP on this segment.
00224 // This is only applicable (but essential) for segments having
00225 // more than one IP on it. It is not the usual situation, so not
00226 // computational intensive.
00227 template <typename V>
00228 static inline bool assign_next_points(V& intersection_points,
00229             segment_identifier const& seg_id)
00230 {
00231     typedef typename boost::range_value<V>::type ip_type;
00232     typedef typename boost::range_const_iterator<V>::type iterator_type;
00233     typedef indexed_source_segment<ip_type> indexed_type;
00234     typedef typename ip_type::traversal_vector vector_type;
00235     typedef typename boost::range_const_iterator<vector_type>::type tvit_type;
00236 
00237     // Create a copy of all IP's on this segment from this source
00238 
00239     std::vector<indexed_type> copy;
00240     copy.reserve(intersection_points.size());
00241     std::size_t index = 0;
00242     for (iterator_type it = boost::begin(intersection_points);
00243          it != boost::end(intersection_points);
00244          ++it, ++index)
00245     {
00246         bool to_be_copied = false;
00247         for (tvit_type tvit = boost::begin(it->info);
00248              !to_be_copied && tvit != boost::end(it->info);
00249              ++tvit)
00250         {
00251             if (tvit->seg_id == seg_id)
00252             {
00253                 to_be_copied = true;
00254             }
00255         }
00256 
00257         if (to_be_copied)
00258         {
00259             // Copy this row, plus ONLY the related information
00260             copy.push_back(indexed_type(index, *it, seg_id));
00261         }
00262     }
00263 
00264     // Normally there are more elements in "copy".
00265     // But in case of merges there could be only one.
00266     if (boost::size(copy) <= 1)
00267     {
00268         return false;
00269     }
00270 
00271     std::sort(copy.begin(), copy.end(), sort_on_distance<indexed_type>());
00272 
00273 
00274     // Now that it is sorted, do the main purpose: assign the next points
00275     typedef typename boost::range_iterator
00276         <
00277             std::vector<indexed_type>
00278         >::type indexed_iterator_type;
00279 
00280     indexed_iterator_type it = boost::begin(copy);
00281     for (indexed_iterator_type prev = it++; it != boost::end(copy); prev = it++)
00282     {
00283         for (
00284 #ifdef GGL_DEBUG_INTERSECTION
00285         typename boost::range_iterator<vector_type>::type
00286 #else
00287         tvit_type
00288 #endif
00289             tvit = boost::begin(it->subject.info);
00290              tvit != boost::end(it->subject.info);
00291              ++tvit)
00292         {
00293             if (tvit->seg_id == seg_id)
00294             {
00295                 assign_next_ip_index(intersection_points, prev->index, seg_id, it->index);
00296 
00297 #ifdef GGL_DEBUG_INTERSECTION
00298                 tvit->next_ip_index = it->index;
00299 #endif
00300 
00301             }
00302         }
00303     }
00304 
00305 #ifdef GGL_DEBUG_INTERSECTION
00306     std::cout << "Enrichment - sorted (on distance, " << seg_id << "): " << std::endl;
00307     report_indexed(copy);
00308 #endif
00309 
00310     return true;
00311 }

00315 // on that segment
00316 template <typename M, typename V>
00317 static inline bool assign_next_points(M& map, V& intersection_points)
00318 {
00319     bool assigned = false;

00321     {
00322         // IF there are more IP's on this segment
00323         if (mit->second > 1)
00324         {
00325             if (assign_next_points(intersection_points, mit->first))

00362              ++tvit)
00363         {
00364             if (tvit->seg_id.source_index == source_index
00365                 && tvit->seg_id.multi_index == multi_index
00366                 && tvit->seg_id.ring_index == ring_index)
00367             {
00368                 to_be_copied = true;
00369             }
00370         }
00371 
00372         if (to_be_copied)
00373         {
00374             // Copy this row, plus ONLY the related information
00375             copy.push_back(indexed_type(index, *it, source_index, multi_index, ring_index));
00376         }
00377     }
00378 
00379 #ifdef GGL_DEBUG_INTERSECTION
00380     std::cout << "Enrichment - ordered/copy (on segment "
00381         << " src: " << source_index
00382         << "): " << std::endl;
00383     report_indexed(copy);
00384 #endif
00385 
00386     std::sort(copy.begin(), copy.end(), sort_on_segment_identifier<indexed_type>());

00391     // assign travel-to-vertex/ip index for each IP

00400              tvit != boost::end(it->subject.info);
00401              ++tvit)
00402         {
00403             if (tvit->seg_id.source_index == source_index
00404                 && tvit->seg_id.multi_index == multi_index
00405                 && tvit->seg_id.ring_index == ring_index)
00406             {
00407                 assign_last_vertex(intersection_points, prev->index, source_index,
00408                     tvit->seg_id.segment_index, it->index);
00409             }
00410         }
00411     }
00412 
00413 #ifdef GGL_DEBUG_INTERSECTION
00414     std::cout << "Enrichment - ordered  on segment (src: "
00415         << source_index << "): " << std::endl;

00417 #endif
00418 
00419     return true;
00420 }

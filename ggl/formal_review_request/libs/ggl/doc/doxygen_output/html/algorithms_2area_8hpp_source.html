00067 {
00068     typedef typename coordinate_type<Box>::type return_type;

00070     static inline return_type apply(Box const& box, Strategy const&)
00071     {
00072         // Currently only works for 2D Cartesian boxes
00073         // Todo: use extreme-strategy
00074         assert_dimension<Box, 2>();
00075 
00076         return_type const dx = get<max_corner, 0>(box) 
00077                 - get<min_corner, 0>(box);
00078         return_type const dy = get<max_corner, 1>(box) 
00079                 - get<min_corner, 1>(box);

00102 template<typename Ring, typename Strategy>
00103 struct ring_area<Ring, clockwise, Strategy>
00104 {
00105     BOOST_CONCEPT_ASSERT( (ggl::concept::AreaStrategy<Strategy>) );
00106 
00107     typedef typename Strategy::return_type type;
00108 
00109     static inline type apply(Ring const& ring, Strategy const& strategy)
00110     {
00111         assert_dimension<Ring, 2>();
00112 
00113         // A closed linear_ring has at least four points,
00114         // if not, there is no (zero) area
00115         if (boost::size(ring) < 4)
00116         {
00117             return type();
00118         }
00119 
00120         typedef typename boost::range_const_iterator<Ring>::type iterator_type;
00121 
00122         typename Strategy::state_type state = strategy.init();
00123 
00124         iterator_type it = boost::begin(ring);
00125         for (iterator_type previous = it++;
00126             it != boost::end(ring);
00127             previous = it++)
00128         {
00129             Strategy::apply(*previous, *it, state);
00130         }
00131         return Strategy::result(state);
00132 
00133     }
00134 };

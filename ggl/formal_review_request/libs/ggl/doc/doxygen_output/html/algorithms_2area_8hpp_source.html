00120     typedef typename S::return_type type;

00122     {
00123         assert_dimension<R, 2>();
00124 
00125         // A closed linear_ring has at least four points,
00126         // if not, there is no (zero) area
00127         if (boost::size(ring) >= 4)

00129             typename S::state_type state_type;

00102 template<typename R, typename S>
00103 struct ring_area
00104 {
00105     typedef typename S::return_type type;
00106     static inline type apply(R const& ring, S const& strategy)
00107     {
00108         assert_dimension<R, 2>();
00109 
00110         // A closed linear_ring has at least four points, if not there is no area
00111         if (boost::size(ring) >= 4)
00112         {
00113             typename S::state_type state_type;

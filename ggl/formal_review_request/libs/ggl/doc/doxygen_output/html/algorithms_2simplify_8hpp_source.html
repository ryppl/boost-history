00052 template<typename R, typename OutputIterator, typename S>
00053 inline void simplify_range_strategy(R const& range, OutputIterator out, double max_distance, S const& strategy)
00054 {
00055     if (boost::begin(range) == boost::end(range) || max_distance < 0)
00056     {
00057         std::copy(boost::begin(range), boost::end(range), out);
00058     }
00059     else
00060     {
00061         typename boost::range_const_iterator<R>::type it = boost::begin(range) + 1;
00062         if (it == boost::end(range) || it + 1 == boost::end(range))
00063         {
00064             std::copy(boost::begin(range), boost::end(range), out);
00065         }
00066         else
00067         {
00068             strategy.simplify(range, out, max_distance);
00069         }
00070     }
00071 }

00092 template<typename R, typename OutputIterator, typename S>
00093 inline void simplify_ring(R const& r_in, OutputIterator out, double max_distance, S const& strategy)
00094 {
00095     // Call do_container for a ring
00096 
00097     // The first/last point (the closing point of the ring) should maybe be excluded because it
00098     // lies on a line with second/one but last. Here it is never excluded.
00099 
00100     // Note also that, especially if max_distance is too large, the output ring might be self intersecting
00101     // while the input ring is not, although chances are low in normal polygons
00102 
00103     // Finally the inputring might have 4 points (=correct), the output < 4(=wrong)
00104     if (boost::size(r_in) <= 4 || max_distance < 0)
00105     {
00106         std::copy(boost::begin(r_in), boost::end(r_in), out);
00107     }
00108     else
00109     {
00110         simplify_range_strategy(r_in, out, max_distance, strategy);
00111     }
00112 }

00114 template<typename Y, typename S>
00115 inline void simplify_polygon(Y const& poly_in, Y& poly_out, double max_distance, S const& strategy)
00116 {
00117     typedef typename boost::range_iterator
00118         <typename interior_type<Y>::type>::type iterator_type;
00119     typedef typename boost::range_const_iterator
00120         <typename interior_type<Y>::type>::type const_iterator_type;
00121 
00122     poly_out.clear();
00123 
00124     // Note that if there are inner rings, and distance is too large, they might intersect with the
00125     // outer ring in the output, while it didn't in the input.
00126     simplify_ring(exterior_ring(poly_in), std::back_inserter(exterior_ring(poly_out)),
00127                   max_distance, strategy);
00128 
00129     interior_rings(poly_out).resize(boost::size(interior_rings(poly_in)));
00130 
00131     const_iterator_type it_in = boost::begin(interior_rings(poly_in));
00132     iterator_type it_out = boost::begin(interior_rings(poly_out));
00133 
00134     for (; it_in != boost::end(interior_rings(poly_in)); it_in++, it_out++)
00135     {
00136         simplify_ring(*it_in, std::back_inserter(*it_out), max_distance, strategy);
00137     }
00138 }

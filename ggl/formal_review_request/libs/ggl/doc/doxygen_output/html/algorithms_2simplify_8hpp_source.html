00093 template<typename Range, typename Strategy, std::size_t Minimum>
00094 struct simplify_range
00095 {
00096     static inline void apply(Range const& range, Range& out,
00097                     double max_distance, Strategy const& strategy)
00098     {
00099         // Call do_container for a linestring / ring
00100 
00101         // For a RING:
00102             // The first/last point (the closing point of the ring) should maybe be excluded because it
00103             // lies on a line with second/one but last. Here it is never excluded.
00104 
00105             // Note also that, especially if max_distance is too large, the output ring might be self intersecting
00106             // while the input ring is not, although chances are low in normal polygons
00107 
00108             // Finally the inputring might have 4 points (=correct), the output < 4(=wrong)
00109 
00110         if (boost::size(range) <= int(Minimum) || max_distance < 0.0)
00111         {
00112             simplify_copy<Range, Strategy>::apply
00113                 (
00114                     range, out, max_distance, strategy
00115                 );
00116         }
00117         else
00118         {
00119             simplify_range_inserter<Range, Strategy>::apply
00120                 (
00121                     range, std::back_inserter(out), max_distance, strategy
00122                 );
00123         }
00124     }
00125 };

00127 template<typename Polygon, typename Strategy>
00128 struct simplify_polygon
00129 {
00130     static inline void apply(Polygon const& poly_in, Polygon& poly_out,
00131                     double max_distance, Strategy const& strategy)
00132     {
00133         typedef typename ring_type<Polygon>::type ring_type;
00134 
00135         typedef typename boost::range_iterator
00136             <typename interior_type<Polygon>::type>::type iterator_type;
00137         typedef typename boost::range_const_iterator
00138             <typename interior_type<Polygon>::type>::type const_iterator_type;
00139 
00140         // Note that if there are inner rings, and distance is too large,
00141         // they might intersect with the outer ring in the output,
00142         // while it didn't in the input.
00143         simplify_range<ring_type, Strategy, 4>::apply(exterior_ring(poly_in),
00144                         exterior_ring(poly_out),
00145                         max_distance, strategy);
00146 
00147         interior_rings(poly_out).resize(boost::size(interior_rings(poly_in)));
00148 
00149         iterator_type it_out = boost::begin(interior_rings(poly_out));
00150 
00151         for (const_iterator_type it_in = boost::begin(interior_rings(poly_in));
00152             it_in != boost::end(interior_rings(poly_in));
00153             ++it_in, ++it_out)
00154         {
00155             simplify_range<ring_type, Strategy, 4>::apply(*it_in,
00156                         *it_out, max_distance, strategy);
00157         }
00158     }

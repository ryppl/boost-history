00043 static inline void get_extremes(const Range& range,
00044             RangeIterator& min_it, RangeIterator& max_it,
00045             const Strategy& strategy)
00046 {
00047     min_it = boost::begin(range);
00048     max_it = boost::begin(range);
00049 
00050     for (RangeIterator it = boost::begin(range) + 1; it != boost::end(range); ++it)

00156     {
00157         typedef typename boost::range_const_iterator<Range>::type range_iterator;
00158 
00159         // Polygons with three corners, or closed with 4 points, are always convex
00160         if (boost::size(range) <= 3)
00161         {
00162             for (range_iterator it = boost::begin(range);
00163                         it != boost::end(range);
00164                         ++it)
00165             {
00166                 m_upper_hull.push_back(*it);

00179         assign_range(range, left_it, right_it, lower_points, upper_points);

00221     inline void build_half_hull(const container& input, container& output,

00232     template <int Factor>
00233     inline void add_to_hull(const P& p, container& output)
00234     {
00235         typename strategy_side<cs_tag, P>::type side;
00236 
00237         output.push_back(p);
00238         register std::size_t output_size = output.size();
00239         while (output_size >= 3)
00240         {
00241             rev_iterator rit = output.rbegin();
00242             const P& last = *rit++;
00243             const P& last2 = *rit++;
00244 
00245             if (Factor * side.side(*rit, last, last2) <= 0)
00246             {
00247                 // Remove last two points from stack, and add last again
00248                 // This is much faster then erasing the one but last.
00249                 output.pop_back();
00250                 output.pop_back();
00251                 output.push_back(last);
00252                 output_size--;
00253             }
00254             else
00255             {
00256                 return;
00257             }
00258         }
00259     }

00235     template <int Factor>
00236     inline void add_to_hull(const P& p, container& output)
00237     {
00238         typename strategy_side<cs_tag, P>::type side;
00239         boost::ignore_unused_variable_warning(side);
00240 
00241         output.push_back(p);
00242         register std::size_t output_size = output.size();
00243         while (output_size >= 3)
00244         {
00245             rev_iterator rit = output.rbegin();
00246             const P& last = *rit++;
00247             const P& last2 = *rit++;
00248 
00249             if (Factor * side.side(*rit, last, last2) <= 0)
00250             {
00251                 // Remove last two points from stack, and add last again
00252                 // This is much faster then erasing the one but last.
00253                 output.pop_back();
00254                 output.pop_back();

00257             }

00259             {
00260                 return;

00297 struct assign<box_tag, Box, 2>
00298 {
00299     typedef typename coordinate_type<Box>::type coordinate_type;
00300 
00301     // Here we assign 4 coordinates to a box.
00302     // -> Most logical is: x1,y1,x2,y2
00303     // In case the user reverses x1/x2 or y1/y2, we could reverse them (THAT IS NOT IMPLEMENTED)
00304 
00305     // Note also comment in util/assign_box_corner ->
00306     //   ("Most logical is LOWER, UPPER and sub-order LEFT, RIGHT")
00307     //   (That is assigning 4 points from a box. So lower-left, lower-right, upper-left, upper-right)
00308     template <typename T>
00309     static inline void apply(Box& box,
00310                 T const& x1, T const& y1, T const& x2, T const& y2)
00311     {
00312         set<min_corner, 0>(box, boost::numeric_cast<coordinate_type>(x1));
00313         set<min_corner, 1>(box, boost::numeric_cast<coordinate_type>(y1));
00314         set<max_corner, 0>(box, boost::numeric_cast<coordinate_type>(x2));

00034 template <typename Ring, typename SegmentIdentifier, typename RangeOut>
00035 struct copy_segments_ring
00036 {
00037     static inline void apply(Ring const& ring,
00038             SegmentIdentifier const& seg_id, int to_index,
00039             RangeOut& current_output)
00040     {
00041         typedef typename boost::range_const_iterator<Ring>::type iterator;
00042 
00043         typedef ggl::ever_circling_iterator<iterator> ec_iterator;
00044 
00045         // The problem: sometimes we want to from "3" to "2" -> end = "3" -> end == begin
00046         // This is not convenient with iterators.
00047 
00048         // So we use the ever-circling iterator and determine when to step out
00049 
00050         int from_index = seg_id.segment_index + 1;
00051 
00052         // Sanity check
00053         BOOST_ASSERT(from_index < boost::size(ring));
00054 
00055         ec_iterator it(boost::begin(ring), boost::end(ring),
00056                     boost::begin(ring) + from_index);
00057 
00058         // [2..4] -> 4 - 2 + 1 = 3 -> {2,3,4} -> OK
00059         // [4..2],size=6 -> 6 - 4 + 2 + 1 = 5 -> {4,5,0,1,2} -> OK
00060         // [1..1], travel the whole ring round
00061         int count = from_index <= to_index
00062             ? to_index - from_index + 1
00063             : boost::size(ring) - from_index + to_index + 1;
00064 
00065         for (int i = 0; i < count; ++i, ++it)
00066         {
00067             // TODO: use 'copy coordinates' to handle different point types
00068 #ifdef GGL_DEBUG_INTERSECTION
00069             std::cout << "  add: ("
00070                 << ggl::get<0>(*it) << ", " << ggl::get<1>(*it) << ")"
00071                 << std::endl;
00072 #endif
00073             current_output.push_back(*it);
00074         }
00075     }
00076 };

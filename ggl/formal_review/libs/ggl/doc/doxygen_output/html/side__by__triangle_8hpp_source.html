00027         // Check at which side of a segment a point lies:
00028         // left of segment (> 0), right of segment (< 0), on segment (0)

00031         {

00040             static inline int apply(P1 const& p1, P2 const& p2, P const& p)
00041             {
00042                 typedef typename select_most_precise
00043                     <
00044                         typename select_most_precise
00045                             <
00046                                 typename coordinate_type<P1>::type,
00047                                 typename coordinate_type<P2>::type
00048                             >::type,
00049                         typename coordinate_type<P>::type
00050                     >::type coordinate_type;
00051 
00052                 coordinate_type const x = get<0>(p);
00053                 coordinate_type const y = get<1>(p);
00054 
00055                 coordinate_type const sx1 = get<0>(p1);
00056                 coordinate_type const sy1 = get<1>(p1);
00057                 coordinate_type const sx2 = get<0>(p2);
00058                 coordinate_type const sy2 = get<1>(p2);
00059 
00060                 coordinate_type const dx = sx2 - sx1;
00061                 coordinate_type const dy = sy2 - sy1;
00062                 coordinate_type const dpx = x - sx1;
00063                 coordinate_type const dpy = y - sy1;
00064 
00065                 coordinate_type s = dx * dpy - dy * dpx;
00066 
00067                 return s > 0 ? 1 : s < 0 ? -1 : 0;
00068             }

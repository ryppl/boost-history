00070     inline void apply(InputRange const& range)
00071     {
00072         // First iterate through this range
00073         // (this two-stage approach avoids many point copies,
00074         //  because iterators are kept in memory. Because iterators are
00075         //  not persistent (in MSVC) this approach is not applicable
00076         //  for more ranges together)
00077 
00078         RangeIterator left_it = boost::begin(range);
00079         RangeIterator right_it = boost::begin(range);
00080 
00081         for (RangeIterator it = boost::begin(range) + 1;
00082             it != boost::end(range);
00083             ++it)
00084         {
00085             if (less(*it, *left_it))
00086             {
00087                 left_it = it;
00088             }
00089 
00090             if (greater(*it, *right_it))
00091             {
00092                 right_it = it;
00093             }
00094         }
00095 
00096         // Then compare with earlier
00097         if (first && boost::size(range) > 0)
00098         {
00099             // First time, assign left/right
00100             left = *left_it;
00101             right = *right_it;
00102             first = false;
00103         }
00104         else
00105         {
00106             // Next time, check if this range was left/right from
00107             // the extremes already collected
00108             if (less(*left_it, left))
00109             {
00110                 left = *left_it;
00111             }
00112 
00113             if (greater(*right_it, right))
00114             {
00115                 right = *right_it;
00116             }
00117         }
00118     }

00314     template <int Factor>
00315     static inline void add_to_hull(point_type const& p, container_type& output)
00316     {
00317         typedef typename strategy_side<cs_tag>::type side;
00318 
00319         output.push_back(p);
00320         register std::size_t output_size = output.size();
00321         while (output_size >= 3)
00322         {
00323             rev_iterator rit = output.rbegin();
00324             point_type const& last = *rit++;
00325             point_type const& last2 = *rit++;
00326 
00327             if (Factor * side::apply(*rit, last, last2) <= 0)
00328             {
00329                 // Remove last two points from stack, and add last again
00330                 // This is much faster then erasing the one but last.
00331                 output.pop_back();
00332                 output.pop_back();
00333                 output.push_back(last);
00334                 output_size--;
00335             }
00336             else
00337             {
00338                 return;

00340         }

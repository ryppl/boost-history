00330         // "Abuse" rangetype for ringtype: for polygon, it is the type of the
00331         // exterior ring. For ring, it is the ring itself. That is what is
00332         // wished here as well.

00335         container_type ips; // intersection points
00336 
00337         bool trivial = ggl::get_intersection_points(geometry1, geometry2, ips);
00338 
00339         // If there are no IP-s, check if one point is in other geometry
00340         // (both geometries have points, see check above)
00341         // If yes, take the correct one
00342 
00343         // (which depends on intersection/union)
00344         // If both are unrelated, take none (intersection) or both (union)
00345 
00346         // If there are disjoint or containing holes:
00347         // - Hole of A can be within hole of B
00348         //   - for union: take hole of A
00349         //   - for intersection: take hole of B
00350         // - Hole of A can be disjoint from B
00351         //   - for union: remove those holes
00352         //   - for intersection: take all those holes
00353         // -> generic procedure
00354         // - Collect all holes in one vector
00355         // - Sort them on decreasing area
00356         // - Check if one is within a larger one,
00357         //   - if yes, take one of them
00358         //   - if no, take all, or none of them

00364         get_not_intersecting_holes<tag2>::apply(geometry2, 1, ips, ni_holes);

00366         if (ips.size() <= 0)
00367         {
00368             GeometryOut result;
00369             if (take_one<Direction>(geometry1, geometry2, result))
00370             {
00371                 add_holes
00372                     <
00373                         tag_out, Geometry1, GeometryOut, hole_vector_type,

00382                     <
00383                         tag_out,
00384                         Geometry2, GeometryOut, hole_vector_type,
00385                         Direction
00386                     >::apply(geometry2, result, ni_holes);

00405 
00406                 // If all IP's are removed, they are all collinear or forming
00407                 // an angle which each other.

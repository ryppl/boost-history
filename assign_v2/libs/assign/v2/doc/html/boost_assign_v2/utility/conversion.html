Convert

Header 

#include <boost/assign/v2/utility/conversion/convert.hpp>

Description

Function converting a range to a container

Syntax 

namespace convert_tag{

    struct put{};
    struct copy{};

}// convert_tag
    template<typename C, typename R>
    C convert(R const& r);

    template<typename C>
    struct convert_adapter;
    
    template<typename R, typename C>
    C operator|( R const& r, convert_adapter<C> );

}// conversion_aux
using conversion_aux::convert;
namespace result_of{

    template<typename C, typename R>
    struct convert;

}//result_of

template<typename C>
conversion_aux::convert_adapter<C> _convert();

Container

Range

Semantics

Expressions

Expression

Description

convert<C>( r, convert_tag::put() )

C cont; put( cont )( as_arg_list( r ) ) and returns cont

convert<C>( r, convert_tag::copy() )

C( begin( r ), end( r ) )

convert<C>( r )

convert<C>( r , tag ), where tag = convert_tag::put if C is either an array or an adapter, tag = convert_tag::copy otherwise.

r | _convert<C>()

convert<C>( r )

Example 

std::vector<int> r( 3 ); r[0] = 72; r[1] = 31; r[2] = 48;
typedef array<int, 3> ar_; ar_ const& ar = ( as2::csv_deque( 72, 31, 48 ) | as2::_convert<ar_>() );

BOOST_ASSIGN_V2_CHECK( ar.front() == 72 );
BOOST_ASSIGN_V2_CHECK( ar.back() == 48 );

Converter

Header 

#include <boost/assign/v2/utility/conversion/converter.hpp>

Description

Wraps a range with container-convertibility

Syntax 

namespace conversion_aux{

    template<typename R>
    class converter
    {

        public:

        explicit converter(typename call_traits<R>::param_type r);

        template<typename C>
        operator C () const;

        template<typename C>
        C type()const;

    };

    struct converter_adapter;

    template<typename R>
    converter<R> operator|( R const& r, converter_adapter );

}// conversion_aux

    conversion_aux::converter_adapter const _converter;

namespace result_of{

    template<typename R>
    struct converter;

}//result_of

    template<typename R>
    typename result_of::converter<R>::type
    converter(R const& r);

Semantics

Notation

c

Instance of converter( r )

Converter

Expression

Description

c.type<C>()

Calls convert<C>( r )

C cont = c

Equivalent to C cont = c.type<C>()

Invocation

Expression

Description

converter( r )

Instance of conversion_aux::converter<R>

r | _converter

Equivalent to converter( r )

Example 

const int sz = 3; typedef array<int, sz>  row_;
array<row_, sz>  matrix3x3 = converter(
    as2::ref::array
        ( as2::ref::csv_array( 1, 2, 3 ) | as2::_convert<row_>() )
        ( as2::ref::csv_array( 4, 5, 6 ) | as2::_convert<row_>() )
        ( as2::ref::csv_array( 7, 8, 9 ) | as2::_convert<row_>() )
);
for(int i = 0; i < 9; i++)
{
    BOOST_ASSIGN_V2_CHECK( matrix3x3[ i / 3 ][ i % 3 ] == i + 1 );
}

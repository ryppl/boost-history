namespace ref{

    template<typename D, typename T>
    class wrapper_crtp
    {
        public:

        void operator=(implementation-defined& t );
        void swap( wrapper_crtp& that );
        operator T&() const;

        bool operator==( const wrapper_crtp& that )const;
        bool operator<( const wrapper_crtp& that )const;
        bool operator==( typename boost::call_traits<T>::param_type t )const;
        bool operator<( typename boost::call_traits<T>::param_type t )const;
        bool operator>( typename boost::call_traits<T>::param_type t );

        template<class CharT, class Traits>

Models a range of reference-wrappers whose reference is accessible by member function get().

Notation

Semantics

Expression

template<typename R> typename result_of::range_get<R const>::type range_get(R const& r);

Example 

typedef int T; T x, y, z;
std::vector<T> r( 3 ); r[0] = 1; r[1] = 2; r[2] = 0;
boost::copy( r, boost::begin(
        as2::ref::csv_array( x, y, z ) | as2::ref::_get
    )
);
BOOST_ASSIGN_V2_CHECK( x == r[0] );
BOOST_ASSIGN_V2_CHECK( z == r[2] );

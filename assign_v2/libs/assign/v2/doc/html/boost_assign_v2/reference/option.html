boost/assign/v2/option.hpp

Warning

Headers in directory boost/assign/v2/option overload operator%. A header from this directory has to be included after that of the object to be passed as first argument to operator%.

Description

Options to alter the semantics of an interpreter.

Semantics

Expression

Description

interpreter % option1 ... % optionm

Returns an interpreter with a modified type or state

Notation

args...

Arguments passed to interpreter

t

Data member internally generated by invoking interpreter( args... )

cont

Container that interpreter internally modifies

C

Type of cont

Data

Header

boost/assign/v2/option/data.hpp

Description

Data generator option.

Semantics

Option

Default

Data-generation

_data = f

None

t = f(args...)

Example

Example1 

std::vector<double> exponent;
typedef double(*fp)(double);
typedef function<double(double)> f_;
(
    as2::put( exponent ) % ( as2::_data = f_( fp(log10) ) )
)( 1000.0 )( 10.0 )( 10000.0 )( 1.0 )( 100.0 );

double eps = numeric::bounds<double>::smallest();
BOOST_ASSIGN_V2_CHECK( fabs( exponent.front() - 3.0 ) < eps );
BOOST_ASSIGN_V2_CHECK( fabs( exponent.back() - 2.0 ) < eps );

Equivalent to as2::put( exponent )( log10( 1000.0 ) )( log10( 10.0 ) )( log10( 10000.0 ) )( log10( 1.0 ) )( log10( 100.0 ) )

Example2 

using namespace lambda;
int i = 1, k = 1;
BOOST_AUTO(
    factorials, (
        as2::deque<int>( as2::_nil ) % (
            as2::_data = ( var(k) *= ( var(i)++ ) )
        )
    )()()()()()
);

BOOST_ASSIGN_V2_CHECK( range::equal( factorials, as2::csv_deque( 1, 2, 6, 24, 120 ) ) );

Equivalent to factorials( k *= i++ ) for i = 1, ..., 5

boost/assign/v2/option/data.hpp

Description

Modifier option.

Semantics

Option

Default

Modifier

_iterate = arg

arg() increments a counter initalized at zero

cont[ arg() ] = t

Example 

using namespace lambda;
typedef int T; array<T, 4> powers; powers[0] = 1; powers[1] = 10;
int index = 2; ( as2::put( powers ) % ( as2::_iterate = var( index )++ ) )( 100 )( 1000 );

BOOST_ASSIGN_V2_CHECK( range::equal( powers, as2::csv_deque( 1, 10, 100, 1000 ) ) );

Mapped

Header

boost/assign/v2/option/mapped.hpp

Description

Modifier option.

Semantics

Option

Default

Modifier

_mapped = arg

None

cont[ t ] = arg( cont[ t ] )

Example 

using namespace lambda;
typedef std::string str_; typedef std::map<str_, int> C; C year; 
( 
    as2::put( year )( "feb", 28 )( "apr", 30 )( "jun", 30 )( "sep", 30 )( "nov", 30 ) 
        % as2::_key %  ( as2::_mapped = ( _1 = 31 ) ) 
)( "jan" )( "mar" )( "may" )( "jul" )( "aug" )( "oct" )( "dec" );

BOOST_ASSIGN_V2_CHECK( year["jan"] == 31 );
BOOST_ASSIGN_V2_CHECK( year["dec"] == 31 );

Calls year["jan"] = 31 etc.

Description

Option modifier framework

Synopsis 

namespace interpreter_aux{

    template<typename Keyword, typename Arg = keyword_aux::ignore>
    struct option_modifier
    {
        
        option_modifier(){}
        option_modifier(Arg arg);
        
        Arg const& arg()const;

        template<typename Arg1>
        option_modifier<Keyword, Arg1>
        operator=(const Arg1& arg1)const;
        
    };

    template<typename Keyword, typename Arg>
    struct meta_modifier_tag
    {
        template<typename D> 
        struct apply;
    };

namespace result_of{
        
    template<typename D>
    struct option_modifier{

            
        template<typename Keyword, typename Arg>
        struct apply 
        {};
        
    };
            
}// result_of

    template<typename C, typename F, typename Tag, typename D, 
        typename Keyword, typename Arg>
    typename ::boost::mpl::apply2<
    	result_of::option_modifier<D>, Keyword, Arg
    >::type
    operator%(
        interpreter_crtp<C, F, Tag, D> const& lhs,
        option_modifier<Keyword, Arg> const& rhs
    );

}// interpreter_aux

	#define BOOST_ASSIGN_V2_OPTION_MODIFIER_KEYWORD(NAME) unspecified
	#define BOOST_ASSIGN_V2_OPTION_MODIFIER_META_MODIFIER_TAG(NAME, Result) unspecified

A class of modifiers

A part of the modifier's implementation

Metafunction class to be specialized on Keyword

Inherits interpreter_crtp<>

Semantics

Requirement

Specialize, such as to model interpreter_aux::ConceptModifier, 

namespace interpreter_aux{

    template<typename Arg>
    class interpreter_modifier< Result >;

}

Expression

Expression

Description

BOOST_ASSIGN_V2_OPTION_MODIFIER_KEYWORD(NAME)

Defines an empty class interpreter_aux::keyword_NAME; and an associated keyword _keyword_NAME

BOOST_ASSIGN_V2_OPTION_MODIFIER_META_MODIFIER_TAG(NAME, Result)

Specialized metafunction class interpreter_aux::meta_modifier_tag on interpreter_aux::keyword_NAME to return Result

Repeat

Header

boost/assign/v2/option/push_front.hpp

Description

Modifier option.

Semantics

Option

Default

Modifier

_repeat = n

n = 0

Invokes n times the modifier that prevailed before passing this option

Example

Example1 

std::vector<int> cont;
( as2::put( cont ) % ( as2::_repeat = 2  ) )( 72 )( 31 )( 48 );

BOOST_ASSIGN_V2_CHECK( range::equal( cont, as2::csv_deque( 72, 72, 31, 31, 48, 48 ) ) );

Example2 

BOOST_AUTO(
    cont, (
        as2::deque<int>( as2::_nil ) % as2::_push_front % ( as2::_repeat = 2 )
    )( 72 )( 31 )( 48 )
);

BOOST_ASSIGN_V2_CHECK( range::equal( cont, as2::csv_deque( 48, 48, 31, 31, 72, 72 ) ) );

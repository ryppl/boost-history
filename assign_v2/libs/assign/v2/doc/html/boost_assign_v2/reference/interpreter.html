Description

Interpreter crtp

Synopsis 

namespace interpreter_aux{

    template<typename C, typename D>
    struct ConceptDerivedInterpreter1{

        BOOST_CONCEPT_USAGE(ConceptDerivedInterpreter1)
        {
            D derived( cont );
            C& ref = derived.container();
        }

        private:
        static C& cont;

    };

    template<typename C, typename F, typename Tag, typename D>
    struct ConceptDerivedInterpreter2 : ConceptDerivedInterpreter1<C, D>{

        typedef interpreter_aux::interpreter_modifier<Tag> modifier_;

        BOOST_CONCEPT_USAGE(ConceptDerivedInterpreter2)
        {
            D derived( cont, f, m );
        }

        private:
        static C& cont;
        static F const &f;
        static modifier_ m;

    };

    template<
        typename C      // Container, 
        , typename F    // Data generator
        , typename Tag  // Modifier tag
        , typename D    // Derived
    >
    class interpreter_crtp
    {

        public:

        typedef  interpreter_modifier<Tag> 
        modifier_type;

        interpreter_crtp(){}
        explicit interpreter_crtp( F const& f );
        explicit interpreter_crtp( F const& f, modifier_type const& m );

        typedef D const& result_type;

        template<typename R>
        result_type
        operator()( as_arg_list_adapter<R> range )const;


        template<typename...Args>
        result_type operator()( Args&&...args )const;


    };

}// interpreter_aux

Notation

C

Container type

T

Expressions

Metafunction deduce_data_generator<C> returns a generator, variadic in its arguments, of T and T*, for value- and pointer-containers, respectively.

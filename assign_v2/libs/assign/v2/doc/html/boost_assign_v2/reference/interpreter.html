As_arg_list

Header

boost/assign/v2/interpreter/as_arg_list.hpp

Description

Range wrapper that tells an interpreter to process each its element individually

Synopsis 

namespace interpreter_aux{    

    template<
        typename R    // Range
    >
    class as_arg_list_adapter;

namespace result_of{

    template<typename R>
    struct as_arg_list;

}// result_of

    template<typename R>
    typename result_of::as_arg_list<R>::type
    as_arg_list(R& range);

    template<typename R>
    typename result_of::as_arg_list<R const>::type
    as_arg_list(R const& range);

}// interpreter_aux

    using interpreter_aux::as_arg_list;

namespace result_of{    
    using interpreter_aux::result_of::as_arg_list;
}// result_of

Crtp

Header

boost/assign/v2/interpreter/crtp.hpp

Description

A crtp class that exposes a functor interface. Arguments are forwarded to a data-generator whose result is forwarded to a container modifier.

Synopsis 

namespace interpreter_aux{


    template<
        typename D              // Derived
        , typename C            // Container
        , typename F            // Data generator
        , typename MTag         // Modifier tag
        , typename DTag         // Data tag
    >
    class interpreter_crtp
    {

        public:

        typedef  interpreter_modifier<Tag> 
        modifier_type;

        interpreter_crtp(){}
        explicit interpreter_crtp( F const& f );
        interpreter_crtp( F const& f, modifier_type const& m );

        typedef D const& result_type;

        template<typename R>
        result_type
        operator()( as_arg_list_adapter<R> range )const;


        template<typename...Args>
        result_type operator()( Args&&...args )const;

    };

}// interpreter_aux

boost/assign/v2/interpreter/data.hpp

Description

Interpreter's data generator. An interpreter's arguments are forwarded to the data-generator whose task is to generate a suitable data-element.

Synopsis

    struct data_generator{};
}// switch_tag

Modifier

Header

boost/assign/v2/interpreter/modifier.hpp

Description

Interpreter's modifier. The value supplied by the data-generator is forwarded to the interpreter's modifier, whose job is to insert that value in a container.

Synopsis 

namespace modifier_tag{

    // Namespace in which to declare template arguments of interpreter_modifier<>

    struct interpreter_modifier
    {
        // Specialize on Tag to model ConceptModifier 
    };

    template<typename MTag, typename DTag, typename C, typename T>
    struct ConceptModifier
    {

        BOOST_CONCEPT_USAGE(ConceptModifier)
        {
            m.impl( cont, t, DTag() ); 
        }

        private:
        static interpreter_aux::interpreter_modifier<MTag>& m;
        static C& cont;
        static T t;
    };

}//interpreter_aux

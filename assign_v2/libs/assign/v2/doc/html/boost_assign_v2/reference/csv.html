Csv

Header

boost/assign/v2/utility/csv.hpp

Description

Maps an arbitrary functor to the csv form.

Synopsis 

namespace result_of{

    template<typename F, typename V>
    struct csv;

}// nth_result_of

    template<typename F>
    typename result_of::csv<F const&,
        ::boost::mpl::vector0<>
    >::type
    csv( F const& functor);

    template<typename F, typename T, typename... Args>
    typename result_of::csv<F const&,
        typename ::boost::mpl::detail::variadic_vector<T, Args...>::type
    >::type
    csv( F const& functor, T&& t, Args&&... args );

Semantics

Notation

t1, ..., tn

Individual elements of args...

Semantics

Expression

Returns

csv( functor, args... )

functor( t1 ) ... ( tn )

Example

Example 1 

std::deque<int> cont;
as2::csv( as2::put( cont ), 72, 31, 48 );

BOOST_ASSIGN_V2_CHECK( range::equal( cont, as2::csv_deque( 72, 31, 48 ) ) );

Example 2 

BOOST_AUTO(
    cont,
    as2::csv( as2::deque<int>( as2::_nil ) % as2::_push_front, 72, 31, 48 )
);

BOOST_ASSIGN_V2_CHECK( range::equal( cont, as2::csv_deque( 48, 31, 72 ) ) );

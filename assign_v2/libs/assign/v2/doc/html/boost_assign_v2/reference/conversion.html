Description

Explicit conversion.

Synopsis 

namespace convert_tag{

    struct put{};
    struct copy{};

}// convert_tag

    template<typename C>
    struct convert_adapter;
    
    template<typename R, typename C>
    C operator|( R const& r, convert_adapter<C> );

}// conversion_aux
namespace result_of{

    template<
        typename C       // Container
        , typename R     // Range
    >
    struct convert;

}//result_of

template<typename C>
conversion_aux::convert_adapter<C> _convert();

Semantics

Expressions

r | convert<C>() is defined as the switch below:

Case

Statement

C is either an Array or an Adapter

C cont; put( cont )( as_arg_list( r ) ) and returns cont

Otherwise

C( begin( r ), end( r ) )

Example 

std::vector<int> r( 3 ); r[0] = 72; r[1] = 31; r[2] = 48;
typedef array<int, 3> ar_; 
ar_ const& ar = ( as2::csv_deque( 72, 31, 48 ) | as2::_convert<ar_>() );

BOOST_ASSIGN_V2_CHECK( range::equal( ar, as2::csv_deque( 72, 31, 48 ) ) );

Header

Description

Implicit conversion.

Synopsis 

namespace conversion_aux{

    template<typename R>
    class converter
    {

        public:

        explicit converter(typename call_traits<R>::param_type r);

        template<typename C>
        operator C () const;

        template<typename C>
        C type()const;

    };

}// conversion_aux
namespace result_of{

    template<typename R>
    struct converter;

}//result_of

    template<typename R>
    typename result_of::converter<R>::type
    converter(R const& r);

Semantics

Notation

c

Instance of converter( r )

Expressions

Expression

Description

c.type<C>()

Calls r | convert<C>()

C cont = c

Equivalent to C cont = c.type<C>()

Example 

const int sz = 3; typedef array<int, sz>  row_;
array<row_, sz>  matrix3x3 = converter(
    as2::ref::array
        ( as2::ref::csv_array( 1, 2, 3 ) | as2::_convert<row_>() )
        ( as2::ref::csv_array( 4, 5, 6 ) | as2::_convert<row_>() )
        ( as2::ref::csv_array( 7, 8, 9 ) | as2::_convert<row_>() )
);

for(int i = 0; i < 9; i++)
{
    BOOST_ASSIGN_V2_CHECK( matrix3x3[ i / 3 ][ i % 3 ] == i + 1 );
}

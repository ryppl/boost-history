#include <boost/assign/v2/put/ext.hpp>

Description

Extended modifiers i.e. perform more complex tasks than, say, push_back().

Iterate

Header 

#include <boost/assign/v2/put/ext/iterate.hpp>

Syntax 

namespace modifier_tag{ struct iterate; }
namespace put_aux{

    template<> class modifier<modifier_tag::iterate>;

}// put_aux

unspecified _iterate;

Semantics

Expression

Implementation

_iterate = n

cont[ i++ ] = t

where i is initialized to n, if specified, zero otherwise.

Example 

typedef int T; boost::array<T, 3> cont; cont[0] = -1;
( as2::put( cont ) % ( as2::_iterate = 1 ) )( 0 )( 1 );

BOOST_ASSIGN_V2_CHECK( cont[0] == -1 );
BOOST_ASSIGN_V2_CHECK( cont[1] == 0 );
BOOST_ASSIGN_V2_CHECK( cont[2] == 1 );

Repeat

Header 

#include <boost/assign/v2/put/ext/repeat.hpp>

Syntax 

namespace modifier_tag{ template<typename OldTag> struct repeat{}; }
namespace put_aux{

    template<typename OldTag> class modifier< modifier_tag::repeat<OldTag> >;

}// put_aux

unspecified _repeat;

Semantics

Expression

Implementation

_repeat = n

m = n; while(--m) put-implementation

where put-implementation is the implementation prior to passing the parameter.

Example 

std::vector<int> cont;
( as2::put( cont ) % ( as2::_repeat = 2  ) )( -1 )( 0 )( 1 );

BOOST_ASSIGN_V2_CHECK( cont.size() == 6 );
BOOST_ASSIGN_V2_CHECK( cont.front() == -1 );
BOOST_ASSIGN_V2_CHECK( cont.back() == 1 );

Lookup

Header 

#include <boost/assign/v2/put/ext/lookup.hpp>

Syntax 

namespace modifier_tag{ template<typename Arg> struct lookup{}; }
namespace put_aux{

    template<typename Arg> struct modifier<modifier_tag::lookup<Arg> >;
    
}//put_aux

unspecified _lookup;

Semantics

Notation

Symbol

key

Instance of an Associative container's key

Expressions

Expression

Implementation

_lookup = f

cont[ key ] = f( cont[ key ] )

Example 

std::map<std::string, int> cont;
cont["jan"] = 31; cont["feb"] = 29; cont["mar"] = 32;
(
    as2::put( cont ) % ( as2::_lookup = ( lambda::_1 -= 1 ) )
)( "feb" )( "mar" );

BOOST_ASSIGN_V2_CHECK( cont["feb"] == 28 );
BOOST_ASSIGN_V2_CHECK( cont["mar"] == 31 );

Replaces put-adapter's functor with a constructor for T

Replaces put-adapter's functor with a functor returning a pointer for T

Example

Math 

std::vector<double> exponent;
typedef function<double(double)> f_;
(
    as2::put( exponent ) % ( as2::_fun = f_( log10 ) )
)( 1000.0 )( 10.0 )( 10000.0 )( 1.0 )( 100.0 ); 


double eps = numeric::bounds<double>::smallest();
assert(fabs( exponent.front() - 3.0 ) < eps );
assert(fabs( exponent.back() - 2.0 ) < eps );

Equivalent to as2::put( exponent )( log10( 1000.0 ) )( log10( 10.0 ) )( log10( 10000.0 ) )( log10( 1.0 ) )( log10( 100.0 ) )

Recursive 

int i = 1, k = 1;
BOOST_AUTO(
    factorials, (
        as2::deque<int>( as2::_nil ) % (
            as2::_fun = ( var(k) *= ( var(i)++ ) )
        )
    )()()()()()
);

assert(factorials.front() == ( 1 ) );
assert(factorials.back() == ( 120 ) );

Equivalent to factorials( k *= i++ ) for i = 1, ..., 5

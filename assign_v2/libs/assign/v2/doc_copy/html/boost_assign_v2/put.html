As_arg_list

Header 

#include <boost/assign/v2/put/adapter/as_arg_list.hpp>

Description

A function that, invoked with a given range, returns a class wrapped around it that signals that the range's elements are to be treated as a sequence of unary arguments to an adapter.

Synopsis 

namespace result_of{

    template<typename R>
    struct as_arg_list
    { 
        typedef unspecified type; 
    };

}// result_of

    template<typename R>
    typename result_of::as_arg_list<R>::type
    _as_arg_list(R& range);
    
    template<typename R>
    typename result_of::as_arg_list<R const>::type
    _as_arg_list(R const& range);

Crtp

Header 

#include <boost/assign/v2/put/adapter/crtp.hpp>

Description

A CRTP class template with a variadic functor interface that maps arguments to a value and inserts it in a container supplied by the derived class. Each of the mapping and the insertion mechanism, termed a modifier, are specified by template paramaters.

Synopsis 

namespace put_aux{

    template<typename C, typename D>
    struct ConceptAdapter1{

        BOOST_CONCEPT_USAGE(ConceptAdapter1)
        {
            D derived( cont );
            C& ref = derived.container();
        }

        private:
        static C& cont;

    };

    template<typename C, typename F, typename Tag, typename D>
    struct ConceptAdapter2 : ConceptAdapter1<C, D>{

        typedef put_aux::adapter_modifier<Tag> modifier_;

        BOOST_CONCEPT_USAGE(ConceptAdapter2)
        {
            D derived( cont, f, m );
        }

        private:
        static C& cont;
        static F const &f;
        static modifier_ m;

    };

    template<typename C, typename F, typename Tag, typename D>
    class adapter_crtp
    {

        public:

        typedef  adapter_modifier<Tag> 
        modifier_type;

        adapter_crtp(){}
        explicit adapter_crtp( F const& f );
        explicit adapter_crtp( F const& f, modifier_type const& m );

        typedef D const& result_type;

        template<typename R>
        result_type
        operator()( as_arg_list_adapter<R> range )const;


        template<typename...Args>
        result_type operator()( Args&&...args )const;


    };

}// put_aux

Modifier

Header 

#include <boost/assign/v2/put/adapter/modifier.hpp>

Description

A class template with no default implementation, but whose specializations are required to expose a particular interface for modifying a container.

Synopsis 

namespace modifier_tag{}
namespace put_aux{

    template<typename Tag>
    struct adapter_modifier{};

}
namespace put_concept{

    template<typename Tag, typename C, typename T>
    struct Modifier
    {

        BOOST_CONCEPT_USAGE(Modifier)
        {
            m.impl( cont, t ); 
        }

        private:
        static put_aux::adapter_modifier<Tag>& m;
        static C& cont;
        static T t;
    };

}

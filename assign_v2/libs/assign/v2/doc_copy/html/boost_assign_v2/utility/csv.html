Csv

Header 

#include <boost/assign/v2/utility/csv.hpp>

Description

Maps a functor and a sequence of values to unary functor calls over these values

Meta

Syntax 

namespace result_of{
	template<typename F, typename V> struct csv;
}

Parameters

Parameter

Description

F

A functor type (or reference thereof)

V

An mpl-vector of arguments, each suitable in a unary-invocation of F

Semantics

Expression

Value

result_of::csv<F, mpl::vector<T[0],..., T[n - 1]>

Type of[a] f( t[0] ) ... ( t[n - 1] )

[a] By recursive application of BOOST_TYPEOF_TPL

Make

Syntax 

template<typename F, typename Args>
   typename result_of::csv<F const&, mpl::vector<Args...> >::type
   csv( F const& functor, Args&&... args );

Semantics

Expression

Returns

csv( f, t[0], ..., t[n - 1] )

f( t[0] ) ... ( t[n - 1] )

Example 

typedef int T; T x = 1, y = 2, z = 0; std::list<T> cont;
as2::csv( as2::put( cont ) , x, y, z );

BOOST_ASSIGN_V2_CHECK( cont.front() == x );
BOOST_ASSIGN_V2_CHECK( cont.back() == z );

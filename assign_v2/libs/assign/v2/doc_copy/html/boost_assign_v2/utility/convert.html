Dispatch

#include <boost/assign/v2/utility/convert/dispatch.hpp>

Synopsis 

namespace convert_aux{

    template<typename C, typename R>
    C dispatch(R const& r, convert_tag::put)
    {
        C cont;
        return (cont | v2::_put_range( r ) );

    }

    template<typename C, typename R>
    C dispatch(R const& r, convert_tag::copy)
    {
        return C( boost::begin( r ), boost::end( r ) );
    }

}// convert_aux

Description

Implementation 

C cont = converter( r );

Copy constructs cont

Forwards to converter( r ).type<C>() 

converter( r ).type<C>();

Returns an object of type C

Deduces tag_ and forwards to convert_aux::dispatch<C>( cont, tag_() ).

Example 

std::vector<int> v( 3 ); v[0] = 1; v[1] = 2; v[2] = 0;
typedef boost::array<int, 3> ar_;
BOOST_ASSIGN_V2_CHECK(
    ( as2::converter( v ).type<ar_>() )[1] == v[1]
);


std::vector<int> v( 3 ); v[0] = -1; v[1] = 0; v[2] = 1;
std::stack<int> lifo = as2::converter( v );
BOOST_ASSIGN_V2_CHECK( lifo.top() == 1 );

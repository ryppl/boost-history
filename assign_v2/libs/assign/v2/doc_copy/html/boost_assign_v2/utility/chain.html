Chain

Header 

#include <boost/assign/v2/utility/chain.hpp>

Description

A variant of join() that caters to reference arrays

Syntax 

namespace chain_aux{

    template<typename R1, typename R2, typename Tag = use_default> struct use_const;

    template<typename Tag1 = use_default, typename Tag2 = void_>
    struct adaptor2
    {

        adaptor2();

        template<typename U>
        struct result{
           typedef unspecified type;
        };

        template<typename R> typename result<R>::type operator()(R& r)const;

        template<typename R> typename result<R const>::type operator()(R const& r)const;

    };

    template<typename R1, typename U2, typename Tag>
    typename chain_aux::result<R1, U2, Tag>::type
    operator|(R1 & r1, chain_aux::adaptor1<U2, Tag> const & h);

    template<typename R1, typename U2, typename Tag>
    typename chain_aux::result<R1 const, U2, Tag>::type
    operator|(R1 const & r1, chain_aux::adaptor1<U2, Tag> const & h);

}// chain_aux

		typedef unspecified type;

Comment

This meta-function becomes relevant if either of r1 or r2 is a reference array: altghouh these are usually constructed in place (rvalues), they may bind to lvalues. Use-const knows not to throw away lvalue-ness, in that case, i.e. returns false.

Example

Read 

typedef std::string T;
array<T, 2> head;  head[0] = "A"; head[1] = "B";
std::list<T> tail; tail.push_back( "C" ); tail.push_back( "D" );
std::vector<T> joined( begin( head ), end( head ) ); copy( tail, std::back_inserter( joined ) );
BOOST_ASSIGN_V2_CHECK(
    range::equal( joined, head | as2::_chain( tail ) )
);

Write 

typedef std::string word; std::vector<word> words( 4 );
words[0] = "foo"; words[1] = "bar"; words[2] = "baz"; 
words[3] = "qux"; words[4] = "quux"; words[5] = "grault"; words[6] = "garply";
array<word, 2> head; std::list<word> tail( 4 );
copy( words, begin( head | as2::_chain( tail ) ) );
BOOST_ASSIGN_V2_CHECK( head[0] == "foo" );
BOOST_ASSIGN_V2_CHECK( head[1] == "baz" );
BOOST_ASSIGN_V2_CHECK( tail.front() == "qux" );
BOOST_ASSIGN_V2_CHECK( tail.back() == "garply" );

Write-to-refs 

using namespace assign::v2;
std::vector<int> consecutive8( 8 ); for(int i = 1; i < 8; i++){ consecutive8[i] = 1 + i; }
array<int, 5> consecutive5; int six, seven, eight;
boost::copy(
    consecutive8,
    begin( consecutive5 && (as2::ref::csv_array( six, seven, eight ) | as2::ref::_get ) )
);

BOOST_ASSIGN_V2_CHECK( consecutive5.front() == 1 );
BOOST_ASSIGN_V2_CHECK( consecutive5.back()  == 5 );
BOOST_ASSIGN_V2_CHECK( six                  == 6 );
BOOST_ASSIGN_V2_CHECK( eight                == 8 );

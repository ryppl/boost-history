Example

Example 1 

typedef std::string str_; typedef std::bitset<3> data_; std::vector<data_> consecutive;
as2::put( consecutive )
    ( str_( "000" ) )( str_( "001" ) )
    ( str_( "010" ) )( str_( "011" ) )
    ( str_( "100" ) )( str_( "101" ) )
    ( str_( "110" ) )( str_( "111" ) );

for(unsigned i = 0; i < consecutive.size(); i++)
{
    assert(  consecutive[i].to_ulong() == i );
}

Keep in mind that although data_( str_( "011" ) ), for instance, is valid, consecutive.push_back( str_( "011" ) ) isn't (GCC4.2)

Calls consecutive.push_back( data_( t ) ); for t = str_( "000" ), ..., str_( "111" )]

Example 2 

std::vector<int> numeric( 10 ); iota( numeric, 0 ); typedef std::string str_;
typedef variant< int, str_ > data_; array<data_, 16> numeric_kb;
as2::put( numeric_kb )( "+" )( "-" )( "*" )( "/" )( "=" )( "." )( as2::_as_arg_list( numeric ) );

assert( get<str_>( numeric_kb.front() ) == "+" );
assert( get<int>( numeric_kb.back()  ) == 9 );

Calls numeric_kb[ i ] = data_( t ) for ( i, t ) = ( 0, "+" ), ..., ( 5, "." )

Calls numeric_kb[ 6 + i ] = data_( *( b + i ) ) for i = 0, ..., 9 and b = boost::begin( numeric )

Example 3 

typedef double data_; typedef std::vector<data_> variable_size_;
variable_size_ a( 3 ); a[0] = 0.71; a[1] = 0.63; a[2] = 0.85;
variable_size_ b( 4 ); b[0] = 0.61; b[1] = 0.69; b[2] = 0.92; b[3] = 0.55;
array<variable_size_, 4> ragged;
as2::put( ragged )
    ( begin( a ), end( a ) )
    ( b )
    ( 1, -99.99 )
    ( );

assert( ragged[0].size() == a.size() );
assert( ragged[3].size() == 0        );

Calls ragged.push_back( variable_size_( begin( a ), end( a ) ) )

Calls ragged.push_back( variable_size_( b ) )

Calls ragged.push_back( variable_size_( 1, -99.99 ) )

Calls ragged.push_back( variable_size_( ) )

Example 4 

std::queue<int> adapter; as2::put( adapter )( 72 )( 31 )( 48 );
array<int, 3> array; as2::put( array )( 72 )( 31 )( 48 );
std::set<int> assoc; as2::put( assoc )( 72 )( 31 )( 48 );
std::list<int> seq; as2::put( seq )( 72 )( 31 )( 48 );

Calls adapter.push( t ) for t = 72, 31, 48

Calls array[i] = t for ( i, t ) = ( 0, 72 ), ( 1, 31 ), ( 2, 48 )

Calls assoc.insert( t ) for t = 72, 31, 48

Calls seq.push_back( t ) for t = 72, 31, 48

Example 5 

typedef const char us_state_ [3]; us_state_ ct = "CT", nj = "NJ", ny = "NY";
typedef int area_code_; typedef tuple<us_state_&,  area_code_> data_; std::deque< data_ > tri_state_area;
as2::put( tri_state_area )( ny, 212 )( ny, 718 )( ny, 516 )( ny, 914 )( nj, 210 )( nj, 908 )( nj, 609 )( ct, 203 );

assert( get<0>( tri_state_area.front() ) == ny );
assert( get<1>( tri_state_area.back()  ) == 203 );

Notice the reference

Calls tri_state.push_back( data_( s, c ) ) for ( s, c ) =( ny, 212 )...( ct, 203 )

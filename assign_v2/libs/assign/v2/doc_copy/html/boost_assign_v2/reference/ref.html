Array
Header 

#include <boost/assign/v2/ref/array/array.hpp>

Description

Functor for generating an array of reference wrappers

Syntax 

namespace ref{
namespace nth_result_of{

    template<array_size_type N, typename U>
    struct array;

}// nth_result_of
namespace result_of{

    template<typename U, typename T = U>
    struct array;

    template<typename U>
    struct array<U, keyword_aux::nil>;

}// result_of

    template<typename T>
    typename result_of::array<T, keyword_aux::nil>::type
    array( keyword_aux::nil );

    template<typename T>
    typename result_of::array<T>::type
    array(T& t);

    template<typename T>
    typename result_of::array<T const>::type
    array(T const & t);

}// ref

U& has to be a reference

Meta-function

Semantics

Expressions

Expression

Description

ref::array<U>( _nil )

Returns an empty reference-array

ref::array( t )

Returns a reference-array of one element binding to t

Example

Example 1 

typedef int const T;
T w = 11, x = 72, y = 31, z = 48;
as2::ref::nth_result_of::array<3, T>::type ar3 = as2::ref::array( w )( x )( y );
{
    T& a = ar3.front(); T& b = ar3.back();

    BOOST_ASSIGN_V2_CHECK( &a == &w );
    BOOST_ASSIGN_V2_CHECK( &b == &y );
}
as2::ref::nth_result_of::array<4, T>::type ar4 = ar3( z );
{
    T& a = ar4.front(); T& b = ar4.back();

    BOOST_ASSIGN_V2_CHECK( &a == &w );
    BOOST_ASSIGN_V2_CHECK( &b == &z );
}

Example 2 

typedef int T; T x, y, z; std::vector<T> r( 3 ); r[0] = 72; r[1] = 31; r[2] = 48;
boost::copy( r, begin( as2::ref::array( x )( y )( z ) | as2::ref::_get ) );

BOOST_ASSIGN_V2_CHECK( x == r[0] );
BOOST_ASSIGN_V2_CHECK( z == r[2] );

Example 3 

typedef int T; T x, y, z; as2::ref::array( x )( y )( z ).assign( -1 );

BOOST_ASSIGN_V2_CHECK( x == -1 );
BOOST_ASSIGN_V2_CHECK( z == -1 );

#include <boost/assign/v2/ref/array/csv_array.hpp>

Description

Function generating an array of reference wrappers

Syntax 

namespace ref{
namespace nth_result_of{

    template<array_size_type N, typename U>
    struct csv_array;

}// result_of

    template<typename U>
    typename nth_result_of::csv_array<0, U>::type
    csv_array( keyword_aux::nil );

namespace result_of{

    template<typename U, typename... Args>
    struct csv_array;

}// result_of


    template<typename T, typename...Args>;

    template<typename T, typename...Args>
    typename result_of::csv_array<T const, Args const...>::type
    csv_array( T const& t, Args const&...args );

}// ref

U& has to be a reference

Meta-function

Semantics

Pre-condition

Each of t, args... are of the same type, T.

Expressions

Expression

Description

ref::csv_array( t, args... )

Returns a reference-array binding to t, args.... The type that is bound to is T if all values are non-const lvalues, T const, otherwise.

Example

Example 1 

typedef int const T;
T x = 72, y = 31, z = 48;
typedef as2::ref::nth_result_of::csv_array<3, T>::type ar_;
ar_ ar = as2::ref::csv_array( x, y, z );
{
    T& a = ar[0]; T& b = ar[2];

    BOOST_ASSIGN_V2_CHECK( &a == &x );
    BOOST_ASSIGN_V2_CHECK( &b == &z );
}

Example 2 

typedef int T; T x, y, z;
std::vector<T> r( 3 ); r[0] = 72; r[1] = 31; r[2] = 48;
boost::copy( r, begin( as2::ref::csv_array( x, y, z ) | as2::ref::_get ) );

BOOST_ASSIGN_V2_CHECK( x == r[0] );
BOOST_ASSIGN_V2_CHECK( z == r[2] );

Example 3 

int x, y, z; as2::ref::csv_array( x, y, z ).assign( -1 );

BOOST_ASSIGN_V2_CHECK( x == -1 );
BOOST_ASSIGN_V2_CHECK( z == -1 );

Copy
Header 

#include <boost/assign/v2/ref/wrapper/copy.hpp>

Description

Reference wrapper with copy semantics.

Syntax 

namespace assign_tag{ struct copy{}; }
namespace assign_copy{ typedef assign_tag::copy assign_tag_; }

    template< class T >
    struct wrapper<ref::assign_tag::copy, T>
    :
        public ref::wrapper_crtp< wrapper<ref::assign_tag::copy, T>, T>
    {
        typedef T type;

        ;

        wrapper();
        explicit wrapper( T& r );
        void rebind(T& r );
        typedef T& result_of_get_type;
        T& get() const;
        T* get_pointer() const;

        using wrapper_crtp<wrapper, T>::operator=;
        void assign(typename boost::call_traits<T>::param_type r );

        protected:
        ;
        T* ref_;
        void swap( wrapper& r );
    };

    template<typename T>
    struct copy_wrapper{ typedef ref::wrapper<assign_tag::copy,T> type; };

}// ref

Equivalent to rebind( r )

Semantics: ref_ = &r

Semantics: *ref_ = r

Crtp
Header 

#include <boost/assign/v2/ref/wrapper/crtp.hpp>

Description

CRTP reference wrapper.

Syntax 

namespace ref{

    template<typename D, typename T>
    class wrapper_crtp :
        boost::less_than_comparable1<
            wrapper_crtp<D, T>,
            boost::less_than_comparable2<
                wrapper_crtp<D, T>,
                T,
                boost::equality_comparable1<
                    wrapper_crtp<D, T>,
                    boost::equality_comparable2<
                        wrapper_crtp<D, T>,
                        T
                    >
                >
            >
        >
    {

        typedef typename boost::remove_const<T>::type non_const;
        typedef typename boost::add_const<T>::type const_;

        public:

        D& derived();
        D const & derived()const;

        void operator=(non_const& t );
        void operator=(const_& t );
        void swap( wrapper_crtp& that );
        operator T&() const;
        bool operator==( const wrapper_crtp& that )const;
        bool operator<( const wrapper_crtp& that )const;
        bool operator==( typename boost::call_traits<T>::param_type t )const;
        bool operator<( typename boost::call_traits<T>::param_type t )const;
        bool operator>( typename boost::call_traits<T>::param_type t )const;

        template<class CharT, class Traits>
        friend std::basic_ostream<CharT, Traits>&
        operator<<(std::basic_ostream<CharT, Traits>& os, const wrapper_crtp& w);

    };


    template<typename D, typename T>
    void rebind( wrapper_crtp<D,T>& a, T& t );

    template<typename D, typename T>
    inline void swap( wrapper_crtp<D,T>& l, wrapper_crtp<D,T>& r );

}// ref

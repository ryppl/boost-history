Csv_array
Header

boost/assign/v2/ref/array/csv_array.hpp

Description

The csv-form designates a syntax that is dual to a functor, that is charaterized by an arity parameter, I. One invocation of the csv-form with an argument list of size Ixn is equivalent to n invocations of the functor, each with I arguments. This form has two variants : basic (size equals one) and extended (size is specified by an integer template parameter).

Synopsis

}// nth_result_of

    array(T& t);

}// ref

Get
Header

boost/assign/v2/ref/array/get.hpp

Description

A range adaptor that converts each element of a reference-array (a reference wrapper) to the reference it holds (by invoking member function get()).

Synopsis 

namespace ref{
namespace result_of{

    template<
        typename R // Range of reference-wrappers
    >
    struct transform_get;

}// result_of

    // Transforms rw by member function get()
    template<typename R>
    typename result_of::transform_get<R>::type
    transform_get(R& rw);
    
    template<typename R>
    typename result_of::transform_get<R const>::type
    transform_get(R const& rw );

    struct get_adaptor;

    template<typename R>
    typename result_of::transform_get<R>::type
    operator|(R& rw, get_adaptor);

    template<typename R>
    typename result_of::transform_get<R const>::type
    operator|(R const& rw, get_adaptor);


    get_adaptor const _get;

}// ref

Example 

namespace as2 = boost::assign::v2;
typedef int T; typedef as2::ref::copy_wrapper<T>::type w_;
boost::array<w_, 3> ref_array; int x, y, z;

boost::copy(
    as2::csv_deque( w_( x ), w_( y ), w_( z ) ),
    boost::begin( ref_array )
);
boost::copy(
    as2::csv_deque( 314, 3141, 31415 ),
    boost::begin(
        ref_array | as2::ref::_get
    )
);

BOOST_ASSIGN_V2_CHECK( x == 314 );
BOOST_ASSIGN_V2_CHECK( y == 3141 );
BOOST_ASSIGN_V2_CHECK( z == 31415 );

boost/assign/v2/ref/wrapper/copy.hpp

Description

Reference wrapper with copy semantics

Synopsis 

namespace assign_tag{ struct copy{}; }
namespace assign_copy{ typedef assign_tag::copy assign_tag_; }

    template< class T >
    struct wrapper<ref::assign_tag::copy, T>
    :
        public ref::wrapper_crtp< wrapper<ref::assign_tag::copy, T>, T>
    {
        
        typedef T type;

        wrapper();

        // Initialization equivalent to rebind( r )            
        explicit wrapper( T& r );

        void rebind(T& r );

        typedef T& result_of_get_type;
        T& get() const;

        T* get_pointer() const;

        using wrapper_crtp<wrapper, T>::operator=;

        // Copy semantics i.e. *ptr = r
        void assign(typename boost::call_traits<T>::param_type r );

        protected:
        ;
        T* ptr;
        void swap( wrapper& r );
    };

    template<typename T>
    struct copy_wrapper{ typedef ref::wrapper<assign_tag::copy,T> type; };

}// ref

Example 

typedef val_ const cval_;
typedef as2::ref::wrapper_param<val_>::type param_;
BOOST_MPL_ASSERT(( boost::is_same<param_, cval_> ));
typedef as2::ref::copy_wrapper<param_>::type w_;
{
    val_&& a = 1;
    w_ w( a );
    BOOST_ASSIGN_V2_CHECK( &w.get() == &a );
}
{
    cval_ a = 1;
    w_ w = w_( 1 );
    BOOST_ASSIGN_V2_CHECK( w.get() == a );
}

namespace ref{

    template<typename D, typename T>
    class wrapper_crtp :
        boost::less_than_comparable1<
            wrapper_crtp<D, T>,
            boost::less_than_comparable2<
                wrapper_crtp<D, T>,
                T,
                boost::equality_comparable1<
                    wrapper_crtp<D, T>,
                    boost::equality_comparable2<
                        wrapper_crtp<D, T>,
                        T
                    >
                >
            >
        >
    {

        typedef typename boost::remove_const<T>::type non_const;
        typedef typename boost::add_const<T>::type const_;

        public:

        D& derived();
        D const & derived()const;

        void operator=(non_const& t );
        void operator=(const_& t );
        void swap( wrapper_crtp& that );
        operator T&() const;
        bool operator==( const wrapper_crtp& that )const;
        bool operator<( const wrapper_crtp& that )const;
        bool operator==( typename boost::call_traits<T>::param_type t )const;
        bool operator<( typename boost::call_traits<T>::param_type t )const;
        bool operator>( typename boost::call_traits<T>::param_type t )const;

        template<class CharT, class Traits>
        friend std::basic_ostream<CharT, Traits>&
        operator<<(std::basic_ostream<CharT, Traits>& os, const wrapper_crtp& w);

    };


    template<typename D, typename T>
    void rebind( wrapper_crtp<D,T>& a, T& t );

    template<typename D, typename T>
    inline void swap( wrapper_crtp<D,T>& l, wrapper_crtp<D,T>& r );

}// ref

Header

boost/assign/v2/conversion/convert.hpp

Description

Explicit conversion.

Synopsis 

namespace conversion_aux{

    template<typename C, typename Arg = kwd_nil_>
    struct convert;

    template<typename R, typename C, typename Arg>
    C operator|( R const& r, convert<C, Arg> const& adapter);

}// conversion_aux

    template<typename C, typename Arg = kwd_nil_>
    struct convert
    {

        convert();
        convert(Arg const& arg);

        template<typename Arg1>
        convert<C, Arg1>
        operator^(Arg1 const& arg1);

    };

Semantics

Expressions

r | convert<C>() is defined as the switch below:

Case

Statement

C is either an Array or an Adapter

C cont; csv_put( cont, as_arg_list( r ) ) and returns cont

C is a multi_array

Undefined

Otherwise

Returns C( begin( r ), end( r ) )

If an argument is passed, r | convert<C>( arg ),

Case

Statement

C is either an Array, Adapter or a multi_array

C cont( arg ); csv_put( cont, as_arg_list( r ) ) and returns cont

Otherwise

Undefined

Example

Example 1 

typedef std::stack<int> C;

C lifo; lifo.push( 3 ); lifo.push( 2 ); lifo.push( 1 );

BOOST_ASSIGN_V2_CHECK(
    ( as2::csv_deque( 3, 1, 1 ) | as2::convert<C>() ) < lifo
);

Converter

Header

boost/assign/v2/conversion/converter.hpp

Description

Implicit conversion.

Synopsis 

namespace conversion_aux{

    template<typename R, typename Arg = kwd_nil_>
    class converter
    {

        public:

        explicit converter(typename call_traits<R>::param_type source);

        converter(
            typename call_traits<R>::param_type source,
            Arg const& arg
        );

        // Implicit conversion
        template<typename C>
        operator C() const;

        // Explicit converion
        template<typename C> C type()const;

    };

}// conversion_aux
namespace result_of{

    template<typename R, typename Arg = kwd_nil_>
    struct converter;

}//result_of

    template<typename R>
    typename result_of::converter<R>::type
    converter(R const& r);

    template<typename R, typename Arg>
    typename result_of::converter<R, Arg>::type
    converter(R const& r, Arg const& arg);

Semantics

See section Convert.

Example

Example 1 

std::list<int> source( 3 ); boost::iota( source, 1 );

std::queue<int> fifo = as2::converter( source );

BOOST_ASSIGN_V2_CHECK( fifo.front() == 1 );
BOOST_ASSIGN_V2_CHECK( fifo.back() == 3 );

Example 2 

template<typename C, typename R>
void f(C cont, R const& r);


typedef int T; typedef std::vector<T> R;
R benchmark( 10 ); boost::iota( benchmark, 0 );

as2::result_of::converter<R>::type source( benchmark );

{
    typedef boost::array<T, 10> C;
    f<C>( source, benchmark );
}
{
    typedef std::deque<T> C;
    f<C>( source, benchmark );
}
{
    typedef std::list<T> C;
    f<C>( source, benchmark );
}
{
    typedef std::queue<T> C;
    f<C>( source, benchmark );
}
{
    typedef std::stack<T> C;
    f<C>( source, benchmark );
}
{
    typedef std::vector<T> C;
    f<C>( source, benchmark );
}

Example 3 

typedef boost::multi_array<int, 2> array2_;
typedef array2_::size_type size_;
typedef size_ const dim_;
dim_ dim1 = 3, dim2 = 3;
array2_ array2 = converter(
    as2::csv_deque(-1, +1, -1, +1, -1, +1, -1, +1, -1),
    boost::extents[dim1][dim2]
);

const int benchmark [] = {
    -1, +1, -1,
    +1, -1, +1,
    -1, +1, -1
};
size_ const n = array2.num_elements();
BOOST_ASSIGN_V2_CHECK(
    boost::range::equal(
        boost::make_iterator_range( array2.data(), n + array2.data() ),
        boost::make_iterator_range( benchmark, n + benchmark )
    )
);

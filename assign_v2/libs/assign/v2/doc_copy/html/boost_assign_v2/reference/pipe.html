Examples

Example1 

typedef const char* T; typedef std::string str_; std::deque<T> cont;

BOOST_ASSIGN_V2_CHECK( 
    boost::range::equal(
        cont | as2::_csv_put( "x", "y", "z" ),
        as2::csv_deque<str_>( "x", "y", "z" )
    )
);

"x", "y" and "z" are kept as const char(&)[2]

Example2 

int k = 1; std::list<int> factorials;
using namespace lambda;

BOOST_ASSIGN_V2_CHECK(
    range::equal(
        factorials | ( 
            as2::_csv_put % ( as2::_data = ( var(k) *= _1 ) ) 
        )( 1, 2, 3, 4, 5 ),
        as2::csv_deque<int>( 1, 2, 6, 24, 120 )
    )
);

Equivalent to calling factorials.push_back( k *= ++i )

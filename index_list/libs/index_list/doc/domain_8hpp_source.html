00249           typedef typename std::vector<pair_type> pair_vector_type;
00250           pair_vector_type pair_collection(num_dimensions_);

00275     /** Domain based on (hyper)planes defined by fixed Manhattan distances.
00276      *
00277      *  Assuming the ordering \f$\mathbf{x}=[x_{d-1},x_{d-2},\ldots x_1,x_0]\f$
00278      *  for the indices and defining \f$s_0, s_1, \ldots, s_{d-1}\f$ such that
00279      *  \f$s_i = \sum_{i = 0}^{d-1} x_i\f$, the effect of reduce() corresponds
00280      *  to the sum \f$\sum_{j=0}^d {s_j + j \choose j+1}\f$. Each possible
00281      *  binomial contribution is tabulated at construction. For reduce() alone,
00282      *  producing a table would not have been necessary. However, expand()
00283      *  would be very slow without it.

00412         /// Query: reduce an IndexCollection to a single index_type.
00413         template< typename IndexCollection >

00422             ((

00450             ascending_[ordering_[num_dimensions_-1]]             ?

00455           {

00458               ascending_[ordering_[i]] ?
00459               index_bases_[ordering_[i]] + s[i] - s[i+1] :

Class template enum_type_cons

boost::enums::enum_type_cons — scoped enum type with constructors

Synopsis

// In header: <boost/enums/scoped/enum_type_cons.hpp>

template<typename ScopedEnum, typename UT = int> 
class enum_type_cons {
public:
  // types
  typedef ScopedEnum::type type;             // c++98 enum type 
  typedef UT               underlying_type;  // underlying type 

  // construct/copy/destruct
  enum_type_cons();
  explicit enum_type_cons(underlying_type);
  enum_type_cons(type);
  enum_type_cons& operator=(enum_type_cons);
  enum_type_cons& operator=(type);

  // public member functions
  operator type();
  type enum_value() const;
  underlying_type underlying_value() const;

  // public static functions
  static enum_type_cons default_value();
  static enum_type_cons convert_to(underlying_type);
  static enum_type_cons convert_to(type);

  // friend functions
  friend underlying_type 
  convert_to(enum_type_cons, 
             boost::dummy::type_tag< underlying_type > const &);
  friend type convert_to(enum_type_cons, 
                         boost::dummy::type_tag< type > const &);
};

Description

param ScopedEnum : Struct scoping the enum.

Macro BOOST_CTS_LIT_LENGTH

BOOST_CTS_LIT_LENGTH — This macro expands into the length of the given string literal without the terminating null character. The "returned" length is a compile-time constant.

Synopsis

BOOST_CTS_LIT_LENGTH(STRING_LITERAL)

Description

This macro is defined as follows: 

#define BOOST_CTS_LIT_LENGTH(STRING_LITERAL) \
	((sizeof(BOOST_CTS_LIT(STRING_LITERAL))/sizeof(::boost::cts::bchar))-1)

Examples
#include <assert.h>
#include <boost/char_type_switch/string.hpp>

int main(void)
{
	assert(BOOST_CTS_LIT_LENGTH("foo") == 3);
	return 0;
}

#include <assert.h>
#include <boost/char_type_switch/string.hpp>

int main(void)
{
	assert(
		BOOST_CTS_LIT_LENGTH("foo") == 
		::boost::cts::bstrlen(BOOST_CTS_LIT("foo"))
	);
	return 0;
}

#include <assert.h>
#include <boost/char_type_switch/string.hpp>

int main(void)
{
	using namespace ::boost::cts;
	bchar foo[BOOST_CTS_LIT_LENGTH("foo")+1];
	bstrcpy(foo, BOOST_CTS_LIT("foo"));
	assert(bstrcmp(foo, BOOST_CTS_LIT("foo")) == 0);
	return 0;
}

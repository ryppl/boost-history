Class template begin

boost::mirror::begin — Returns the iterator pointing to the first element of the MetaObjectSequence.

Synopsis

template<typename MetaObjectSequence> 
class begin {
public:
  // types
  typedef unspecified type;
};

Description

Examples
typedef ::std::pair< bool, double> P;
BOOST_CHECK((
	deref<
		begin< BOOST_MIRRORED_CLASS(P)::attributes >::type
	>::type::base_name() == 
	at< 
		BOOST_MIRRORED_CLASS(P)::attributes, 
		mpl::int_<0> 
	>::type::base_name()
));

typedef ::std::pair< bool, double> P;
BOOST_CHECK((
	deref<
		begin< BOOST_MIRRORED_CLASS(P)::all_attributes >::type
	>::type::base_name() == 
	BOOST_CTS_LIT("first")
));

typedef ::std::pair< bool, double> P;
BOOST_MPL_ASSERT((
	iterator_equal<
		begin< BOOST_MIRRORED_CLASS(P)::attributes >::type,
		point_to< 
			BOOST_MIRRORED_CLASS(P)::attributes, 
			mpl::int_<0>
		>::type
	>
));
BOOST_MPL_ASSERT((
	iterator_not_equal<
		begin< BOOST_MIRRORED_CLASS(P)::all_attributes >::type,
		point_to< 
			BOOST_MIRRORED_CLASS(P)::all_attributes, 
			mpl::int_<1>
		>::type
	>
));

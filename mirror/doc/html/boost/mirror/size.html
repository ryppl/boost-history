Class template size

boost::mirror::size — Returns the size of the MetaObjectSequence.

Synopsis

template<typename MetaObjectSequence> 
class size : public MplIntegralConstant {
};

Description

Examples
size_t x = size< 
		BOOST_MIRRORED_CLASS(
			::boost::tuple< 
				bool, 
				char, 
				wchar_t, 
				short int, 
				int, 
				long int, 
				float, 
				double
			>
		)::attributes
	>::value;

BOOST_MPL_ASSERT_RELATION((
	size< BOOST_MIRRORED_CLASS(::std::pair< int, int>)::attributes>::value,
	==,
	2
));

BOOST_MPL_ASSERT_RELATION((
	size< BOOST_MIRRORED_CLASS(::std::pair< int, int>)::all_attributes>::value,
	==,
	2
));

Class template reflects_type

boost::mirror::reflects_type — If MetaObject is either an instantiation of meta_type or meta_class, then reflects_type inherits from ::boost::true_type. Otherwise it inherits from ::boost::false_type.

Synopsis

template<typename MetaObject> 
class reflects_type : public false_type-or-true_type {
};

Description

This template can be used to distinguish whether the opaque MetaObject type reflects a type or a class and thus is an instantiation of meta_type.

Examples
BOOST_STATIC_ASSERT((
	reflects_type< BOOST_MIRRORED_TYPE(::std::string) >::value
));

BOOST_STATIC_ASSERT((
	reflects_type< BOOST_MIRRORED_CLASS(::std::pair<int, ::std::string>) >::value
));

BOOST_STATIC_ASSERT((
	!reflects_type< BOOST_MIRRORED_NAMESPACE(::boost::mirror) >::value
));

BOOST_STATIC_ASSERT((
	!reflects_type< BOOST_MIRRORED_TYPE(::std::string)::scope >::value
));

BOOST_MPL_ASSERT((
	reflects_type< BOOST_MIRRORED_TYPE(::std::string) >
));

BOOST_MPL_ASSERT_NOT((
	reflects_type< BOOST_MIRRORED_NAMESPACE(::boost::mirror) >
));

Class template iterator_less

boost::mirror::iterator_less — If the Iterator1 is less than Iterator2 then iterator_less<Iterator1, Iterator2> inherits from ::boost::mpl::true_. Otherwise it inherits from ::boost::mpl::false_.

Synopsis

template<typename Iterator1, typename Iterator2> 
class iterator_less : public mpl::false_-or-mpl::true_ {
};

Description

Examples
typedef ::std::pair<int, int> P;
typedef BOOST_MIRRORED_CLASS(P) meta_P;
BOOST_MPL_ASSERT((
	iterator_less<
		point_to<
			meta_P::attributes,
			mpl::int_<0>
		>::type,
		point_to<
			meta_P::attributes,
			mpl::int_<1>
		>::type
	>
));
BOOST_MPL_ASSERT((
	iterator_less<
		begin<meta_P::all_attributes>::type,
		end<meta_P::all_attributes>::type
	>
));

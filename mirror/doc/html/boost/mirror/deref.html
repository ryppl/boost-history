Class template deref

boost::mirror::deref — Returns the MetaObject that the Iterator points to.

Synopsis

template<typename Iterator> 
class deref {
public:
  // types
  typedef MetaObject type;
};

Description

Examples
BOOST_CHECK((
	at< 
		BOOST_MIRRORED_CLASS(
			::std::pair< bool, double>
		)::attributes, 
		mpl::int_<0> 
	>::type::base_name() == 
	deref<
		point_to<
			BOOST_MIRRORED_CLASS(
				::std::pair< bool, double>
			)::attributes, 
			mpl::int_<0> 
		>::type
	>::type::base_name()
));

Class template is_valid

boost::mirror::is_valid — If the Iterator is valid then is_valid<Iterator> inherits from ::boost::mpl::true_. Otherwise it inherits from ::boost::mpl::false_.

Synopsis

template<typename Iterator> 
class is_valid : public mpl::false_-or-mpl::true_ {
};

Description

Examples
typedef ::boost::tuple<int, int, int> T;
typedef BOOST_MIRRORED_CLASS(T) meta_T;
BOOST_MPL_ASSERT((
	is_valid<
		begin<meta_T::attributes>::type
	>
));
BOOST_MPL_ASSERT_NOT((
	is_valid<
		end<meta_T::attributes>::type
	>
));
BOOST_MPL_ASSERT((
	is_valid<
		rbegin<meta_T::all_attributes>::type
	>
));
BOOST_MPL_ASSERT_NOT((
	is_valid<
		rend<meta_T::all_attributes>::type
	>
));

Class template at

boost::mirror::at — Returns the i-th element of the MetaObjectSequence. The Position parameter must be an instantiation of mpl::int_.

Synopsis

template<typename MetaObjectSequence, typename Position> 
class at {
public:
  // types
  typedef unspecified type;
};

Description

Examples
BOOST_CHECK((
	at< 
		BOOST_MIRRORED_CLASS(
			::std::pair< bool, double>
		)::attributes, 
		mpl::int_<0> 
	>::type::base_name() == 
	BOOST_CTS_LIT("first")
));

));

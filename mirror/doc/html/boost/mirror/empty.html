Class template empty

boost::mirror::empty — If MetaObjectSequence is empty then empty<MetaObjectSequence> inherits from ::boost::mpl::true_. Otherwise it inherits from ::boost::mpl::false_.

Synopsis

template<typename MetaObjectSequence> 
class empty : public mpl::false_-or-mpl::true_ {
};

Description

Examples
BOOST_MPL_ASSERT_NOT((
	empty< BOOST_MIRRORED_CLASS(::std::pair< int, int>)::attributes >
));

BOOST_MPL_ASSERT_NOT((
	empty< BOOST_MIRRORED_CLASS(::std::pair< int, int>)::all_attributes >
));

Class template reflects_namespace

boost::mirror::reflects_namespace — If MetaObject is an instantiation of meta_namespace, then reflects_namespace inherits from ::boost::true_type. Otherwise it inherits from ::boost::false_type.

Synopsis

template<typename MetaObject> 
class reflects_namespace : public false_type-or-true_type {
};

Description

This template can be used to distinguish whether the opaque MetaObject type reflects a namespace and thus is an instantiation of meta_namespace.

Examples
BOOST_STATIC_ASSERT((
	reflects_namespace< BOOST_MIRRORED_NAMESPACE(::boost::mirror) >::value
));

BOOST_STATIC_ASSERT((
	reflects_namespace< BOOST_MIRRORED_TYPE(::std::string)::scope >::value
));

BOOST_STATIC_ASSERT((
	!reflects_namespace< BOOST_MIRRORED_TYPE(::std::string) >::value
));

BOOST_MPL_ASSERT((
	reflects_namespace< BOOST_MIRRORED_NAMESPACE(::boost::mirror) >
));

BOOST_MPL_ASSERT((
	reflects_namespace< BOOST_MIRRORED_TYPE(::std::string)::scope >
));

BOOST_MPL_ASSERT_NOT((
	reflects_namespace< BOOST_MIRRORED_TYPE(::std::string) >
));

The following code sample shows the registering and reflection of namespaces, and the use of the reflects_namespace trait template. 

#include <boost/mirror/meta_namespace.hpp>
#include <boost/mirror/traits/reflects_namespace.hpp>
#include <boost/mpl/assert.hpp>


// define some namespaces
namespace foo {
namespace bar {
} // namespace bar
} // namespace foo

namespace baz {
} // namespace baz

// register them with mirror
namespace boost {
  namespace mirror {
    // ::foo
    BOOST_MIRROR_REG_NAMESPACE((foo))
    // ::foo::bar
    BOOST_MIRROR_REG_NAMESPACE((foo)(bar))
    // ::baz
    BOOST_MIRROR_REG_NAMESPACE((baz))
} // namespace mirror
} // namespace boost    


int main(void)
{
  using ::boost::mirror::reflects_namespace;
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::boost::mirror) >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::foo) >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::foo)::scope >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::foo::bar) >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::foo::bar)::scope >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::foo::bar)::scope::scope >
  ));
  BOOST_MPL_ASSERT((
    reflects_namespace< BOOST_MIRRORED_NAMESPACE(::baz) >
  ));
  return 0;
}

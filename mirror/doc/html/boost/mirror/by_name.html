Class template by_name

boost::mirror::by_name — Allows to get the individual MetaAttributes in the MetaClassAttributes which was passed as a parameter, by the name of the attributes they are reflecting.

Synopsis

template<typename MetaClassAttributes> 
class by_name : public unspecified {
};

Description

Examples
typedef ::std::pair< bool, double> P;
typedef BOOST_MIRRORED_CLASS(P) meta_P;
BOOST_CHECK((
	by_name<meta_P::attributes>::first::base_name() == 
	BOOST_CTS_LIT("first")
));
BOOST_CHECK((
	by_name<meta_P::attributes>::second::base_name() == 
	BOOST_CTS_LIT("second")
));
BOOST_CHECK((
	by_name<meta_P::all_attributes>::first::base_name() == 
	at<meta_P::all_attributes, mpl::int_<0> >::type::base_name()
));
BOOST_CHECK((
	by_name<meta_P::all_attributes>::second::base_name() == 
	at<meta_P::all_attributes, mpl::int_<1> >::type::base_name()
));

typedef ::std::pair< bool, double> P;
typedef BOOST_MIRRORED_CLASS(P) meta_P;
BOOST_MPL_ASSERT((
	is_same<
		by_name<meta_P::all_attributes>::first::type::reflected_type,
		bool
	>
));
BOOST_MPL_ASSERT((
	is_same<
		by_name<meta_P::all_attributes>::second::type::reflected_type,
		double
	>
));

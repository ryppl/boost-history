Class template rend

boost::mirror::rend — Returns a reverse iterator pointing before the first element of the MetaObjectSequence.

Synopsis

template<typename MetaObjectSequence> 
class rend {
public:
  // types
  typedef unspecified type;
};

Description

Examples
typedef ::std::pair< int, ::std::string> P;
BOOST_MPL_ASSERT((
	iterator_not_equal<
		rbegin< BOOST_MIRRORED_CLASS(P)::attributes >::type,
		rend< BOOST_MIRRORED_CLASS(P)::attributes >::type
	>
));

Class template iterator_equal

boost::mirror::iterator_equal — If the Iterator1 is equal to Iterator2 then iterator_equal<Iterator1, Iterator2> inherits from ::boost::mpl::true_. Otherwise it inherits from ::boost::mpl::false_.

Synopsis

template<typename Iterator1, typename Iterator2> 
class iterator_equal : public mpl::false_-or-mpl::true_ {
};

Description

Examples
typedef ::std::pair<int, int> P;
typedef BOOST_MIRRORED_CLASS(P) meta_P;
BOOST_MPL_ASSERT((
	iterator_equal<
		begin<meta_P::attributes>::type,
		point_to<
			meta_P::attributes,
			mpl::int_<0>
		>::type
	>
));
BOOST_MPL_ASSERT_NOT((
	iterator_equal<
		begin<meta_P::all_attributes>::type,
		end<meta_P::all_attributes>::type
	>
));

Type definition bostream

bostream — This type is defined as ::std::ostream if bchar is char and is defined as ::std::wostream if bchar is wchar_t.

Synopsis

typedef ::std::basic_ostream<bchar, bchar_traits> bostream;

Description

The bostream type is defined as an abbreviation for ::std::basic_ostream<bchar, bchar_traits>.

Examples
#include <assert.h>
#include <boost/char_type_switch/string.hpp>
#include <boost/char_type_switch/iostream.hpp>

int main(void)
{
	using namespace ::std;
	using namespace ::boost::cts;
	bistream& in  = bcin();
	bostream& out = bcout();

	bstring str;
	out << BOOST_CTS_LIT("Enter a string: ") << flush;
	in >> str;
	out << BOOST_CTS_LIT("The string is ") << 
		str.length() << 
		BOOST_CTS_LIT(" characters long") << 
	endl;
	return 0;
}

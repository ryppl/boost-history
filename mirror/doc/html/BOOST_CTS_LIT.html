Macro BOOST_CTS_LIT

BOOST_CTS_LIT — This macro expands into wide character string literal if the BOOST_CTS_USE_WIDE_CHARS symbol is defined, or expands into char string literal otherwise.

Synopsis

BOOST_CTS_LIT(STRING_LITERAL)

Description

This macro is defined as follows: 


#ifdef BOOST_CTS_USE_WIDE_CHARS
#	define BOOST_CTS_LIT(STRING_LITERAL) L##STRING_LITERAL
#else
#	define BOOST_CTS_LIT(STRING_LITERAL) STRING_LITERAL
#endif

Examples
#include <boost/char_type_switch/string.hpp>
//...

#ifdef BOOST_CTS_USE_WIDE_CHARS
BOOST_MPL_ASSERT((
	::boost::is_same<
		BOOST_TYPEOF(BOOST_CTS_LIT("foo")),
		BOOST_TYPEOF(L"foo")
	>
));
#else
BOOST_MPL_ASSERT((
	::boost::is_same<
		BOOST_TYPEOF(BOOST_CTS_LIT("foo")),
		BOOST_TYPEOF("foo")
	>
));
#endif

#include <assert.h>
#include <boost/char_type_switch/string.hpp>

int main(void)
{
#ifdef BOOST_CTS_USE_WIDE_CHARS
	assert(::boost::cts::bstrcmp(BOOST_CTS_LIT("foo"), L"foo") == 0);
#else
	assert(::boost::cts::bstrcmp(BOOST_CTS_LIT("foo"), "foo") == 0);
#endif
	return 0;
}

#include <assert.h>
#include <boost/char_type_switch/string.hpp>

int main(void)
{
	assert(::boost::cts::bstrcmp(BOOST_CTS_LIT("foo"), BOOST_CTS_STRINGIZE(foo)) == 0);
	return 0;
}

free-functions

  1. bistream& bcin();

    Returns a reference to the standard input stream.

    If bchar is defined as char then this function returns a reference to ::std::cin, otherwise if bchar is defined as wchar_t, then bcin returns a reference to ::std::wcin. 

  2. bostream& bcout();

    Returns a reference to the standard output stream.

    If bchar is defined as char then this function returns a reference to ::std::cout, otherwise if bchar is defined as wchar_t, then bcout returns a reference to ::std::wcout.

free-functions

  1. size_t bstrlen(const bchar* str);

    Returns the length (the count of characters) of the given string without the terminating null character.

    Calling this function is equivalent to calling strlen if bchar is defined as char or to calling wcslen if bchar is defined as wchar_t.

    Examples
    #include <assert.h>
#include <boost/char_type_switch/cstring.hpp>

int main(void)
{
	using namespace ::boost::cts;
	assert(bstrlen(BOOST_CTS_LIT("foo")) == 3);
	assert(bstrlen(BOOST_CTS_LIT("foo")) == BOOST_CTS_LIT_LENGTH("foo"));
	//
	const bchar bar[] = BOOST_CTS_LIT("bar");
	assert(bstrlen(bar) == 3);
	return 0;
}

  2. int bstrcmp(const bchar* str1, const bchar* str2);

    This function compares the given two strings and returns an integral value that indicates the relationship between them.

    Calling this function is equivalent to calling strcmp if bchar is defined as char or to calling wcscmp if bchar is defined as wchar_t. If the first string is lexicographically less than the second the return value is negative, if the first string is lexicographically greater than the second the return value is positive, if the strings are equal then the return value is zero.

    Examples
    #include <assert.h>
#include <boost/char_type_switch/cstring.hpp>

int main(void)
{
	using namespace ::boost::cts;
	const bchar foo[] = BOOST_CTS_LIT("foo");
	const bchar bar[] = BOOST_CTS_LIT("bar");

	assert(bstrcmp(foo, bar) > 0);
	assert(bstrcmp(foo, foo) == 0);
	assert(bstrcmp(bar, foo) < 0);

	assert(bstrcmp(foo, BOOST_CTS_LIT("baz")) > 0);
	assert(bstrcmp(foo, BOOST_CTS_STRINGIZE(foobar)) <= 0);
	return 0;
}

  3. bchar* bstrcpy(bchar* dst, const bchar* src);

    This function copies src into dst including the terminating null character and returns dst. dst should not overlap src in memory.

    Calling this function is equivalent to calling strcpy if bchar is defined as char or to calling wcscpy if bchar is defined as wchar_t.

    Examples
    #include <assert.h>
#include <boost/char_type_switch/cstring.hpp>

int main(void)
{
	using namespace ::boost::cts;
	bchar aaa[4];
	bchar bbb[4];

	bstrcpy(aaa, BOOST_CTS_LIT("foo"));
	bstrcpy(bbb, BOOST_CTS_LIT("bar"));
	assert(bstrcmp(aaa, bbb) != 0);

	bstrcpy(bbb, aaa);
	assert(bstrcmp(aaa, bbb) == 0);
	return 0;
}

  4. bchar* bstrncpy(bchar* dst, const bchar* src, size_t num);

    This function copies the first num characters from src into dst and returns dst. If the end of src is found before num characters are copied, dst is padded with null characters until the count of modified characters reaches num. If num is less or equal to the length of src, then the null character is not implicitly appended to dst.

    Calling this function is equivalent to calling strncpy if bchar is defined as char or to calling wcsncpy if bchar is defined as wchar_t.

    Examples
    #include <assert.h>
#include <boost/char_type_switch/cstring.hpp>

int main(void)
{
	using namespace ::boost::cts;
	const size_t size = 4;
	bchar aaa[size];
	bchar bbb[size];

	bstrncpy(aaa, BOOST_CTS_LIT("foo"), size);
	bstrncpy(bbb, BOOST_CTS_LIT("bar"), size);
	assert(bstrcmp(aaa, bbb) != 0);

	bstrncpy(bbb, aaa, size);
	assert(bstrcmp(aaa, bbb) == 0);

	bstrncpy(bbb, BOOST_CTS_LIT("foobar"), size);
	bbb[size-1] = BOOST_CTS_LIT('\0');
	assert(bstrcmp(aaa, bbb) == 0);	return 0;
}

Asynchronous completion token

Handle

The asynchronous-completion-token dispatches processing actions (_tasks_) in response to the completion of asynchronous operations. asynchronous-completion-token uniquely identifies the task and state necessary to process the result of the operation [3] .

Handle

boost::task::handle represents an asynchronous-completion-token. It will be returned by boost::task::async() and is associated with the submitted boost::task::task.

:act_interface Asynchronous completion token interface

boost::task::handle implements an interface in order to check the state of computation and result transfer

  • get_id(): return id of the associated task-id
  • is_ready(): test if result is set
  • has_value(): test if value-result is set
  • has_exception(): test if exception-result is set
  • get(): return value-result
  • wait(): wait until result is set
  • wait_for(): wait until result is set or time-duration has elapsed
  • wait_until(): wait until result ist set or time-point has elapsed 

long fibonacci( long n)
{
	boost::this_thread::interruption_point(); // interruption point
	
	if ( n == 0) return 0;
	if ( n == 1) return 1;
	long k1( 1), k2( 0);
	for ( int i( 2); i <= n; ++i)
	{
		if ( boost::this_thread::interruption_requested() )	// check if interruption was requested
			return;

		long tmp( k1);
		k1 = k1 + k2;
		k2 = tmp;
	}
	
	boost::this_thread::interruption_point();  // interruption point
	
	return k1;
}

void main()
{
	boost::task::handle< long > h(
		boost::task::async(
			boost::task::new_thread(),
			boost::task::make_task(
				fibonacci,
				10) ) );
	std::cout << "id == " << h.get_id() << "\n";
	std::cout << "is ready == " << std::boolalpha << h.is_ready() << "\n";
	h.wait();
	std::cout << "has value == " << std::boolalpha << h.has_value() << "\n";
	std::cout << "has exception == " << std::boolalpha << h.has_exception() << "\n";
	std::cout << "fibonacci(10) == " << h.get() << std::endl;
}

:task_interruption Task interruption
  • interrupt(): interrupt boost::task::task and return immediately
  • __fn_interrupt_andwait_: interrupt and wait until boost::task::task was removed from worker-thread
  • interrupt_and_wait_for(): interrupt and wait until boost::task::task was removed from worker-thread or time duration has elapsed
  • interrupt_and_wait_until(): interrupt and wait until boost::task::task was removed from worker-thread or time point has elapsed
  • boost::this_thread::interruption_requested(): return bool if interruption was requested 

long cooperative( long n)
{
	boost::this_thread::interruption_point(); // interruption point
	
	if ( n == 0) return 0;
	if ( n == 1) return 1;
	long k1( 1), k2( 0);
	for ( int i( 2); i <= n; ++i)
	{
		if ( boost::this_thread::interruption_requested() )	// check if interruption was requested
			return;
			
		long tmp( k1);
		k1 = k1 + k2;
		k2 = tmp;
	}
	
	boost::this_thread::interruption_point();  // interruption point
	
	return k1;
}

void main()
{
	boost::task::handle< long > h(			// get handle associated with the task
		boost::task::async(			
			boost::task::new_thread(),	// asynchronous executor
			boost::task::make_task(		// task, to be executed asynchronously
				cooperative,
				10) ) );
	h.interrupt_and_wait();
	std::cout << "id == " << h.get_id() << "\n";
	std::cout << "is ready == " << std::boolalpha << h.is_ready() << "\n";
	std::cout << "has value == " << std::boolalpha << h.has_value() << "\n";
	std::cout << "has exception == " << std::boolalpha << h.has_exception() << "\n";
	std::cout << h.get() << std::endl;		// throws boost::task::task_interrupted
}

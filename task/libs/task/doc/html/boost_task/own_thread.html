Execute in own thread

boost::task::own_thread executes the task in the current thread (synchronous execution - concerns of N2802 do not aply). 

long fibonacci( long n)
{
	if ( n == 0) return 0;
	if ( n == 1) return 1;
	long k1( 1), k2( 0);
	for ( int i( 2); i <= n; ++i)
	{
		long tmp( k1);
		k1 = k1 + k2;
		k2 = tmp;
	}
	return k1;
}

void main()
{
	boost::task::task< long > t( fibonacci, 10);

	boost::task::handle< long > h(
		boost::task::async(
			boost::move( t),
			boost::task::own_thread() ) );

	std::cout << "fibonacci(10) == " << h.get() << std::endl;
}
